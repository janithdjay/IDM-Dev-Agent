from pathlib import Path

def print_directory_tree(root_dir: Path, current_dir: Path, prefix: str = "", ignore_dirs=None):
    if ignore_dirs is None:
        # Explicitly omit environments, caches, and raw data indexing paths
        ignore_dirs = {
            ".git", 
            "__pycache__", 
            "agent_env", 
            "venv", 
            ".venv", 
            "node_modules", 
            "data", 
            "index", 
            "index_data"
        }
        
    try:
        # Gather and sort files so folders appear cleanly structured
        paths = sorted(list(current_dir.iterdir()), key=lambda p: (p.is_file(), p.name.lower()))
    except PermissionError:
        # Handle system folders or restricted files gracefully
        return

    for i, path in enumerate(paths):
        if path.name in ignore_dirs:
            continue
            
        is_last = (i == len(paths) - 1)
        connector = "└── " if is_last else "├── "
        
        print(f"{prefix}{connector}{path.name}")
        
        # Recurse if the path is a directory
        if path.is_dir():
            new_prefix = prefix + ("    " if is_last else "│   ")
            print_directory_tree(root_dir, path, new_prefix, ignore_dirs)

if __name__ == "__main__":
    project_root = Path(".")
    print(f"\n📁 {project_root.resolve().name}/")
    print_directory_tree(project_root, project_root)

import json
from pathlib import Path
from typing import Dict, Any
from dataclasses import dataclass, asdict


@dataclass
class TaskThresholds:
    """Configuration for task generation thresholds"""
    # Fan-out thresholds (function calls too many others)
    fan_out_medium: int = 5
    fan_out_high: int = 10
    fan_out_critical: int = 20
    
    # Fan-in thresholds (function called by too many)
    fan_in_medium: int = 5
    fan_in_high: int = 10
    fan_in_critical: int = 20
    
    # LLM task generation settings
    llm_enabled: bool = True
    llm_max_tasks: int = 10
    llm_model: str = "qwen2.5-coder:3b"


class TaskConfig:
    """Configuration manager for task generation"""
    
    def __init__(self, project_id: str):
        self.project_id = project_id
        self.config_dir = Path("data/config")
        self.config_file = self.config_dir / f"{project_id}_task_config.json"
        self._ensure_dir()
        self.thresholds = self._load()
    
    def _ensure_dir(self):
        self.config_dir.mkdir(parents=True, exist_ok=True)
    
    def _load(self) -> TaskThresholds:
        """Load config from disk or create default"""
        if self.config_file.exists():
            try:
                data = json.loads(self.config_file.read_text(encoding="utf-8"))
                return TaskThresholds(**data)
            except Exception as e:
                print(f"Error loading config: {e}")
                return TaskThresholds()
        return TaskThresholds()
    
    def save(self) -> None:
        """Save config to disk"""
        self.config_file.write_text(
            json.dumps(asdict(self.thresholds), indent=2),
            encoding="utf-8"
        )
    
    def get(self) -> TaskThresholds:
        """Get current thresholds"""
        return self.thresholds
    
    def update(self, updates: Dict[str, int]) -> None:
        """Update thresholds and save"""
        for key, value in updates.items():
            if hasattr(self.thresholds, key):
                setattr(self.thresholds, key, value)
        self.save()
    
    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary for API response"""
        return asdict(self.thresholds)

# Add is_public_api method to FunctionDetail
def is_public_api(self) -> bool:
    """Check if function is likely a public API"""
    # Not starting with underscore (unless it's __init__ etc)
    if self.name.startswith('_') and not self.name.startswith('__'):
        return False
    # Has docstring often indicates public API
    if self.docstring:
        return True
    # Is a method without underscore prefix
    return not self.name.startswith('_')

FunctionDetail.is_public_api = is_public_api

from fastapi import APIRouter, Query
from fastapi.responses import StreamingResponse
from pathlib import Path

from backend.services.context.context_builder import ContextBuilder

from backend.services.llm.ollama_client import OllamaClient
from backend.services.project_service import ProjectService
from config import INDEX_DIR

router = APIRouter(prefix="/api/agent", tags=["Agent"])

# FIXED: Use consistent index path resolution
project = ProjectService().get_active_project()
INDEX_PATH = INDEX_DIR / project["project_id"]

# Instantiate services
context_builder = ContextBuilder(INDEX_PATH)
prompt_builder = PromptBuilder()
llm_client = OllamaClient()


@router.get("/stream")
async def stream_agent_response(
    question: str = Query(..., description="The user query or code question"),
    symbol_name: str = Query(None, description="Target symbol if explicitly extracted")
):
    """
    Asynchronously streams repository context analysis back to the development chat frame.
    """
    def token_generator():
        context_data = None
        intent = "explain_symbol"

        if symbol_name:
            context_data = context_builder.build(symbol_name, intent="explain_symbol")

        if not context_data:
            intent = "general_query"
            context_data = {
                "message": "No direct codebase symbol matches resolved for this request context."
            }

        full_prompt = prompt_builder.build(
            intent=intent,
            crawler_review_data=context_data,
            question=question,
            history=[]
        )

        for token in llm_client.generate_stream(full_prompt):
            yield token

    return StreamingResponse(token_generator(), media_type="text/plain")

def get_project_id() -> str:
    """Get current project ID"""
    project_service = ProjectService()
    active_project = project_service.get_active_project()
    if not active_project:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="No active project selected"
        )
    return active_project["project_id"]


@router.get("/analyze/function/{function_name}", response_model=FunctionDetailResponse)
async def analyze_function(
    function_name: str,
    project_id: str = Depends(get_project_id)
):
    """
    Get detailed information about a function including parameters and callers.
    """
    analyzer = CodeAnalyzer(project_id)
    func_detail = analyzer.get_function_detail(function_name)
    
    if not func_detail:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail=f"Function '{function_name}' not found"
        )
    
    # Create preview (first 200 chars)
    preview = func_detail.source_code[:200] + "..." if len(func_detail.source_code) > 200 else func_detail.source_code
    
    return FunctionDetailResponse(
        name=func_detail.name,
        file_path=func_detail.file_path,
        line_start=func_detail.line_start,
        line_end=func_detail.line_end,
        parameters=[p.to_dict() for p in func_detail.parameters],
        return_type=func_detail.return_type,
        decorators=func_detail.decorators,
        is_async=func_detail.is_async,
        is_method=func_detail.is_method,
        class_name=func_detail.class_name,
        docstring=func_detail.docstring,
        caller_count=len(func_detail.callers),
        source_code_preview=preview
    )


@router.post("/impact/add-parameter", response_model=ImpactAnalysisResponse)
async def analyze_add_parameter_impact(
    request: AddParameterRequest,
    project_id: str = Depends(get_project_id)
):
    """
    Analyze impact of adding a new parameter to a function.

async def search_functions(
    query: str,
    limit: int = 20,
    project_id: str = Depends(get_project_id)
):
    """
    Search for functions by name.
    """
    analyzer = CodeAnalyzer(project_id)
    symbols = analyzer._load_symbols()
    
    results = []
    for sym in symbols:
        if sym.get("type") in ["function", "method"]:
            name = sym.get("name", "")
            if query.lower() in name.lower():
                results.append({
                    "name": name,
                    "file": sym.get("file", ""),
                    "type": sym.get("type"),
                    "parent": sym.get("parent")
                })
                if len(results) >= limit:
                    break
    
    return {"functions": results, "total": len(results)}

@router.post("/tasks/generate")
def generate_tasks() -> dict:
    """Generate new tasks from current codebase analysis"""
    services = _get_services()

    new_tasks = services["task_generator"].generate_tasks()

    existing_tasks = services["task_storage"].load_tasks()
    existing_ids = {t["id"] for t in existing_tasks}

    added = []
    for task in new_tasks:
        if task.id not in existing_ids:
            existing_tasks.append(task.to_dict())
            added.append(task.id)

    # Convert dicts back to AnalysisTask objects
    all_tasks = []
    for t_dict in existing_tasks:
        # Reconstruct evidence
        evidence_objects = []
        for e in t_dict.get("evidence", []):
            evidence_objects.append(TaskEvidence(
                symbol=e["symbol"],
                metric_type=e["metric_type"],
                metric_value=e["metric_value"],
                file_path=e["file_path"],
                line_range=e.get("line_range"),
                callers=e.get("callers", []),
                callees=e.get("callees", [])
            ))
        
        all_tasks.append(AnalysisTask(
            id=t_dict["id"],
            title=t_dict["title"],
            description=t_dict["description"],
            category=TaskCategory(t_dict["category"]),
            priority=TaskPriority(t_dict["priority"]),
            status=TaskStatus(t_dict["status"]),
            evidence=evidence_objects,
            suggested_fix=t_dict.get("suggested_fix"),
            impact_analysis=t_dict.get("impact_analysis"),
            created_at=t_dict["created_at"],
            updated_at=t_dict["updated_at"],
            notes=t_dict.get("notes")
        ))

    services["task_storage"].save_tasks(all_tasks)

    return {
        "message": f"Generated {len(added)} new tasks",
        "total_tasks": len(existing_tasks),
        "new_task_ids": added
    }

    task = None
    for t in tasks:
        if t["id"] == task_id:

            break

    if evidence:
        symbol = evidence[0].get("symbol") if isinstance(evidence[0], dict) else evidence[0].symbol
    else:
        symbol = "unknown"

    chain = services["graph_service"].get_call_chain(symbol, max_depth=2)

    llm = OllamaClient()
    prompt = f"""
You are a senior software engineer. Analyze this task and provide impact analysis.

TASK: {task.get('title')}
DESCRIPTION: {task.get('description')}

EVIDENCE:
- Symbol: {symbol}
- Metric: {evidence[0].get('metric_type') if evidence else 'N/A'}
- Value: {evidence[0].get('metric_value') if evidence else 'N/A'}

CALL CHAIN ANALYSIS:
- Callers (incoming): {chain.get('callers', [])[:10]}
- Callees (outgoing): {chain.get('callees', [])[:10]}

Provide:
1. RISK ASSESSMENT (High/Medium/Low)
2. IMPACT SCOPE (which parts of system affected)
3. RECOMMENDED APPROACH
4. ESTIMATED EFFORT (S/M/L)
"""

    analysis = llm.generate(prompt)

    return {
        "task_id": task_id,
        "impact_analysis": analysis,
        "call_chain": chain
    }

from pathlib import Path
import json


class ProjectService:
    def __init__(self, projects_file=None):
        if projects_file is None:
            self.projects_file = (
                Path(__file__).resolve().parents[2]
                / "data"
                / "projects.json"
            )
        else:
            self.projects_file = Path(projects_file)

        self._projects_cache = None

    def _load_projects(self):
        if self._projects_cache:
            return self._projects_cache

        if not self.projects_file.exists():
            raise FileNotFoundError("projects.json not found")

        data = json.loads(self.projects_file.read_text(encoding="utf-8"))
        self._projects_cache = data["projects"]
        return self._projects_cache

    def get_project(self, project_id: str):
        projects = self._load_projects()

        for p in projects:
            if p["project_id"] == project_id:
                return p

        raise ValueError(f"Project not found: {project_id}")

    def list_projects(self):
        return self._load_projects()
    
    def get_active_project(self):
        projects = self._load_projects()

        for project in projects:
            if project.get("active", False):
                return project

        raise ValueError("No active project found in projects.json")
    
    def get_index_path_for_project(self, project_id: str = None) -> Path:
        """Get the index path for a project"""
        from config import INDEX_DIR
        
        if project_id is None:
            project = self.get_active_project()
            project_id = project["project_id"]
        
        return INDEX_DIR / project_id

class ChatEngine:

    def __init__(self):

        self.intent_classifier = IntentClassifier()

        self.entity_extractor = EntityExtractor()

        self.execution_engine = ExecutionEngine()

        self.memory = ConversationMemory()

        self.reference_resolver = ReferenceResolver()

        self.llm = OllamaClient(
            model="qwen2.5-coder:3b"
        )

    def chat(
        self,
        question: str
    ):

        # ---------------------------------
        # Store user message
        # ---------------------------------

        self.memory.remember(
            role="user",
            content=question
        )

        intent = self.intent_classifier.classify(
            question
        )

        extracted_symbol = self.entity_extractor.extract_symbol(
            question
        )

        if extracted_symbol:

            symbol = extracted_symbol

        else:

            if self.reference_resolver.should_use_memory(
                question
            ):

                symbol = self.memory.get_last_symbol()

            else:

                symbol = None

        print("=" * 60)
        print("Question:", question)
        print("Resolved Symbol:", symbol)
        print("=" * 60)

        # ---------------------------------
        # General chat
        # ---------------------------------

        if symbol is None:

            prompt = f"""
You are an expert software engineering assistant.

Question:

{question}
"""

            answer = self.llm.generate(prompt)

            self.memory.remember(
                role="assistant",
                content=answer
            )

            return {

                "question": question,

                "intent": "general_chat",

                "symbol": None,

                "answer": answer

            }

        # ---------------------------------
        # Project-aware execution
        # ---------------------------------

        result = self.execution_engine.execute(

            intent=intent,

            symbol=symbol,

            question=question,

            history=self.memory.get_recent_history()

        )

        self.memory.remember(

            role="assistant",

            content=result.get("answer"),

            symbol=result.get("symbol"),

            intent=result.get("intent")

        )

        return {

            "question": question,

            "intent": result.get("intent"),

            "symbol": result.get("symbol"),

            "answer": result.get("answer"),

            "context_used": result.get("context_used"),

            "metrics": result.get("metrics")

        }

class ContextCompressor:

    def compress(self, context: dict, intent: str):

        compressed = dict(context)

        if intent == "explain_symbol":
            compressed["calls"] = context.get("calls", [])[:20]
            compressed["related_symbols"] = context.get("related_symbols", [])[:10]

            reasoning = context.get("reasoning", {})

            compressed["reasoning"] = {
                "callers_chain": reasoning.get("callers_chain", [])[:5],
                "callees_chain": reasoning.get("callees_chain", [])[:5],
            }

        elif intent == "find_callers":

            compressed = {
                "symbol": context.get("symbol"),
                "called_by": context.get("called_by", []),
                "reasoning": context.get("reasoning", {}),
            }

        return compressed

from collections import deque


class ConversationMemory:

    def __init__(self, max_messages=20):

        self.max_messages = max_messages

        self.messages = deque(maxlen=max_messages)

        self.last_symbol = None

        self.last_intent = None

    def remember(
        self,
        role: str,
        content: str,
        symbol=None,
        intent=None
    ):

        self.messages.append({

            "role": role,

            "content": content

        })

        if symbol:

            self.last_symbol = symbol

        if intent:

            self.last_intent = intent

    def get_last_symbol(self):

        return self.last_symbol

    def get_last_intent(self):

        return self.last_intent

    def get_recent_history(
        self,
        limit=6
    ):

        return list(self.messages)[-limit:]

    def clear(self):

        self.messages.clear()

        self.last_symbol = None

        self.last_intent = None

import re

from backend.services.project_service import ProjectService
from config import INDEX_DIR
from backend.services.agent.symbol_resolver import SymbolResolver


class EntityExtractor:

    def __init__(self):
        project = ProjectService().get_active_project()
        self.resolver = SymbolResolver(
            INDEX_DIR / project["project_id"]
        )

    def extract_symbol(
        self,
        question: str
    ):

        patterns = [
            r"where\s+is\s+([A-Za-z_][A-Za-z0-9_]*)\(?\)?\s+called",
            r"who\s+calls\s+([A-Za-z_][A-Za-z0-9_]*)",
            r"explain\s+([A-Za-z_][A-Za-z0-9_]*)",
            r"what\s+does\s+([A-Za-z_][A-Za-z0-9_]*)",
            r"can\s+i\s+delete\s+([A-Za-z_][A-Za-z0-9_]*)",
            r"safe\s+to\s+remove\s+([A-Za-z_][A-Za-z0-9_]*)"
        ]

        candidate = None

        for pattern in patterns:

            match = re.search(
                pattern,
                question,
                re.IGNORECASE
            )

            if match:

                candidate = match.group(1)

                break

        # -------------------------
        # Regex match succeeded
        # -------------------------

        if candidate:

            resolved = self.resolver.resolve(
                candidate
            )

            if resolved:

                return resolved

        # -------------------------
        # Fallback scan
        # -------------------------

        return self.resolver.resolve_from_text(
            question
        )

import time


class ExecutionEngine:

    def __init__(self):
        project = ProjectService().get_active_project()
        self.context_builder = ContextBuilder(INDEX_DIR / project["project_id"])
        self.graph_service = CallGraphService(INDEX_DIR / project["project_id"])
        self.prompt_builder = PromptBuilder()
        # self.llm = OllamaClient(model="qwen2.5-coder:7b")
        self.llm = OllamaClient(model="qwen2.5-coder:3b")
        self.reasoning_engine = ReasoningEngine(self.graph_service)
        self.cache = ContextCache(ttl_seconds=300)
        self.compressor = ContextCompressor()
        

    def execute(self, intent: str, symbol: str, question: str = None, history=None):
        start = time.perf_counter()

        # 1. Build cached context
        cache_key = f"{intent}:{symbol}"

        context = self.cache.get(cache_key)

        if context is None:
            context = self.context_builder.build(
                symbol_name=symbol,
                intent=intent
            )

            if context is not None:
                self.cache.set(cache_key, context)

        if context is None:
            return {
                "error": "Symbol not found"
            }

        print("=" * 60)
        print("Intent:", intent)
        print("Context keys:", list(context.keys()))
        print(
            "Context chars:",
            len(str(context))
        )
        print("=" * 60)
        
        t1 = time.perf_counter()
        print(f"Context: {t1 - start:.3f}s")

        reasoning_key = f"reasoning:{symbol}"

        reasoning_context = self.cache.get(reasoning_key)
        if not reasoning_context:
            reasoning_context = self.reasoning_engine.build_reasoning_context(symbol)
            self.cache.set(reasoning_key, reasoning_context)
            
        t2 = time.perf_counter()
        print(f"Reasoning: {t2 - t1:.3f}s")

        context = {
            **context,
            "reasoning": reasoning_context
        }

        # 2. FAST PATH (NO LLM)
        if intent == "find_callers":
            return {
                "symbol": symbol,
                "intent": intent,
                "answer": context.get("called_by", []),
                "context_used": context
            }

        # 3. LLM PATH
        compressed_context = self.compressor.compress(
            context=context,
            intent=intent
        )
        
        prompt = self.prompt_builder.build(
            intent=intent,
            symbol_data=compressed_context,
            question=question,
            history=history or []
        )
        print("=" * 60)
        print("Prompt length:", len(prompt))
        print("Compressed context chars:", len(str(compressed_context)))
        print("=" * 60)
        
        t3 = time.perf_counter()
        print(f"Prompt: {t3 - t2:.3f}s")

        response = self.llm.generate(prompt)

        t4 = time.perf_counter()

        print(f"LLM: {t4 - t3:.3f}s")
        print(f"TOTAL: {t4 - start:.3f}s")
        print("=" * 60)

        return {
            "symbol": symbol,
            "intent": intent,
            "answer": response,
            "context_used": compressed_context,
            "metrics": {
                "context_ms": round((t1 - start) * 1000, 1),
                "reasoning_ms": round((t2 - t1) * 1000, 1),
                "prompt_ms": round((t3 - t2) * 1000, 1),
                "llm_ms": round((t4 - t3) * 1000, 1),
                "total_ms": round((t4 - start) * 1000, 1),
                "context_chars": len(str(context)),
                "prompt_chars": len(prompt),
                "model": "qwen2.5-coder:3b"
            }
        }

import re

from backend.services.agent.query_models import Intent


class IntentClassifier:

    def classify(self, question: str):

        q = question.lower().strip()

        # -----------------------------------
        # Find Callers
        # -----------------------------------

        caller_patterns = [

            r"where.*called",
            r"who.*calls",
            r"called by",
            r"find callers?",
            r"show callers?",
            r"list callers?"

        ]

        for pattern in caller_patterns:

            if re.search(pattern, q):

                return Intent.FIND_CALLERS

        # -----------------------------------
        # Dependency Analysis
        # -----------------------------------

        dependency_patterns = [

            r"depends on",
            r"\buses\b",
            r"\breferences\b",
            r"what.*uses",
            r"what.*references",
            r"dependencies?",
            r"external dependencies?"

        ]

        for pattern in dependency_patterns:

            if re.search(pattern, q):

                return Intent.DEPENDENCY_ANALYSIS

        # -----------------------------------
        # Impact Analysis
        # -----------------------------------

        impact_patterns = [

            r"can.*delete",
            r"safe.*remove",
            r"safe.*delete",
            r"should.*remove",
            r"should.*delete",
            r"can.*refactor",
            r"safe.*modify",
            r"what.*depends",
            r"impact",
            r"what.*break",
            r"what happens if",
            r"remove.*will",
            r"delete.*will"

        ]

        for pattern in impact_patterns:

            if re.search(pattern, q):

                return Intent.IMPACT_ANALYSIS

        # -----------------------------------
        # Explain Symbol
        # -----------------------------------

        explain_patterns = [

            r"what.*does",
            r"explain",
            r"describe",
            r"summarize",
            r"tell me about",
            r"walk me through",
            r"overview",
            r"give me an overview",
            r"how.*works",
            r"how.*implemented",
            r"how.*written",
            r"how.*designed",
            r"purpose of",
            r"what is",
            r"why.*exist",
            r"why.*used"

        ]

        for pattern in explain_patterns:

            if re.search(pattern, q):

                return Intent.EXPLAIN_SYMBOL

        # -----------------------------------
        # Unknown
        # -----------------------------------

        return Intent.UNKNOWN

from typing import Dict, Any, List
from backend.services.agent.prompt_templates.registry import PromptRegistry


class PromptBuilder:
    """
    Central prompt construction layer for the agent.
    """

    def __init__(self):
        pass

    def build(
        self,
        intent: str,
        crawler_review_data: Dict[str, Any],
        question: str,
        history: List[Dict[str, str]] = None
    ) -> str:

        template = PromptRegistry.get(intent)

        base_instructions = self._system_instructions()

        # ---------------------------
        # TEMPLATE HANDLING (NEW CORE LOGIC)
        # ---------------------------

        if template:
            system_block = template.system()
            task_block = template.task()
            format_block = template.output_format()
        else:
            system_block = base_instructions
            task_block = self._fallback_task(intent)
            format_block = ""

        history_block = self._build_history(history or [])

        context_block = self._build_context(crawler_review_data)

        return f"""
{system_block}

{task_block}

{format_block}

# RECENT CONVERSATION

{history_block}

# CURRENT USER QUESTION

{question}

# CODE CONTEXT

{context_block}

# IMPORTANT RULES

- The recent conversation is additional context.
- The current user question has highest priority.
- Use the conversation history to resolve references like:
    - it
    - this
    - that method
    - summarize it
    - explain further
    - make it shorter
- Never invent symbols.
- Only use symbols present in the supplied context.
- If uncertain, say "unknown from context".
"""

    # -------------------------------------------------

    def _system_instructions(self):

        return """
You are a senior software engineering assistant.

You help developers understand a local codebase.

You are a reasoning engine.

You do not invent APIs or functions.

You answer only from supplied context.
"""

    # -------------------------------------------------

    def _fallback_task(self, intent: str):

        if str(intent) == "explain_symbol":

            return "TASK: Explain the requested symbol."

        if str(intent) == "find_callers":

            return "TASK: Identify callers and call chains."

        if str(intent) == "impact_analysis":

            return """
TASK: Assess modification or deletion impact.
Include callers, dependencies, and risks.
"""

        if str(intent) == "dependency_analysis":

            return """
TASK: Analyze dependencies (incoming and outgoing).
"""

        if str(intent) == "codebase_review":
            return """
    TASK: Review codebase structure and suggest improvements.
    Focus on architecture, design, and maintainability.
    """

        return "TASK: General code analysis."

    # -------------------------------------------------

    def _build_history(
        self,
        history
    ):

        if not history:
            return "No previous conversation."

        lines = []

        for item in history:

            role = item.get("role", "unknown")
            content = item.get("content", "")

            lines.append(f"{role.upper()}: {content}")

        return "\n\n".join(lines)

    # -------------------------------------------------

    def _build_context(
        self,
        symbol_data
    ):

        if not symbol_data:
            return "No symbol context found."

        lines = []

        lines.append(f"Symbol: {symbol_data.get('symbol')}")
        lines.append(f"Type: {symbol_data.get('type')}")
        lines.append(f"File: {symbol_data.get('file')}")
        lines.append(f"Parent: {symbol_data.get('parent')}")

        calls = symbol_data.get("calls", [])
        if calls:
            lines.append("\nDirect Calls:")
            lines.append(", ".join(calls[:20]))

        related = symbol_data.get("related_symbols", [])
        if related:
            lines.append("\nRelated Symbols:")
            lines.append(", ".join(related[:10]))

        callers = symbol_data.get("called_by", [])
        if callers:
            lines.append("\nCalled By:")
            for caller in callers[:10]:
                lines.append(str(caller))

        reasoning = symbol_data.get("reasoning", {})
        if reasoning:
            lines.append("\nReasoning:")

            for item in reasoning.get("callers_chain", [])[:10]:
                lines.append(f"{item['from']} -> {item['to']}")

            for item in reasoning.get("callees_chain", [])[:10]:
                lines.append(f"{item['from']} -> {item['to']}")

        return "\n".join(lines)

from collections import deque


class ReasoningEngine:
    """
    Expands symbol context using call graph relationships.
    Enables multi-hop reasoning for debugging questions.
    """

    def __init__(self, graph_service):
        self.graph = graph_service

    def build_reasoning_context(self, symbol: str, max_depth: int = 2):
        """
        Expands:
        - callers
        - callees
        - indirect chains
        """

        visited = set()
        queue = deque([(symbol, 0)])

        callers_map = self.graph.build_graph().get("called_by", {})
        calls_map = self.graph.build_graph().get("calls", {})

        result = {
            "root": symbol,
            "callers_chain": [],
            "callees_chain": []
        }

        while queue:
            node, depth = queue.popleft()

            if node in visited or depth > max_depth:
                continue

            visited.add(node)

            # callers
            for caller in callers_map.get(node, []):
                result["callers_chain"].append({
                    "from": caller,
                    "to": node,
                    "depth": depth + 1
                })
                queue.append((caller, depth + 1))

            # callees
            for callee in calls_map.get(node, []):
                result["callees_chain"].append({
                    "from": node,
                    "to": callee,
                    "depth": depth + 1
                })
                queue.append((callee, depth + 1))

        return result

import re


class ReferenceResolver:

    def __init__(self):

        self.reference_patterns = [

            r"\bit\b",
            r"\bthis\b",
            r"\bthat\b",
            r"\bthis method\b",
            r"\bthat method\b",
            r"\bthis function\b",
            r"\bthat function\b",
            r"\bthis class\b",
            r"\bthat class\b",
            r"\bthe method\b",
            r"\bthe function\b",
            r"\bthe class\b"

        ]

    def should_use_memory(
        self,
        question: str
    ):

        q = question.lower()

        for pattern in self.reference_patterns:

            if re.search(pattern, q):
                return True

        return False

import json
import re
from pathlib import Path


class SymbolResolver:

    def __init__(self, index_path: Path):

        self.index_path = Path(index_path)

        self.symbols = self._load_symbols()

    def _load_symbols(self):

        file = self.index_path / "symbols.json"

        if not file.exists():
            return []

        return json.loads(
            file.read_text(encoding="utf-8")
        )

    def resolve(self, candidate: str):

        if not candidate:
            return None

        candidate = candidate.lower().strip()

        # Exact match

        for symbol in self.symbols:

            if symbol["name"].lower() == candidate:
                return symbol["name"]

        # Ignore spaces

        normalized = candidate.replace(" ", "_")

        for symbol in self.symbols:

            if symbol["name"].lower() == normalized:
                return symbol["name"]

        # Ignore underscores

        normalized = normalized.replace("_", "")

        for symbol in self.symbols:

            if symbol["name"].lower().replace("_", "") == normalized:
                return symbol["name"]

        return None

    def resolve_from_text(self, text: str):

        """
        Fallback scan.

        Looks through every word in the sentence
        and attempts to resolve it as a project symbol.
        """

        tokens = re.findall(
            r"[A-Za-z_][A-Za-z0-9_]*",
            text
        )

        for token in tokens:

            resolved = self.resolve(token)

            if resolved:

                return resolved

        return None

        self.index_service = index_service
        self.graph_service = graph_service

    def build_review_context(self) -> Dict[str, Any]:
        index = self.index_service.get_index()
        graph = self.graph_service.build_graph()

        project_root = getattr(self.graph_service, 'index_path', None)
        if project_root:
            project_root = Path(project_root).parent.parent.parent
        else:
            project_root = Path.cwd()

        return {
            "modules": self._extract_modules(index, project_root),
            "symbols": self._extract_symbols(index),
            "dependencies": self._extract_dependencies(graph),
            "hotspots": self._find_hotspots(graph),
            "architecture_summary": self._build_arch_summary(graph)
        }

    def _extract_modules(self, index: Dict[str, Any], project_root: Path) -> List[str]:
        symbols = index.get("symbols", [])
        modules: Set[str] = set()
        project_root = Path(project_root).resolve()

        for item in symbols:
            if not isinstance(item, dict):
                continue
            file_path = item.get("file")
            if not file_path:
                continue
            try:
                absolute_path = Path(file_path).resolve()
                try:
                    relative = absolute_path.relative_to(project_root)
                    module_parts = relative.with_suffix("").parts
                    if module_parts:
                        module = ".".join(module_parts)
                        modules.add(module)
                    else:
                        modules.add(relative.stem)
                except ValueError:
                    modules.add(str(absolute_path.stem))
            except Exception:
                modules.add(Path(file_path).stem)

        return sorted(modules)

    def _extract_symbols(self, index: Dict[str, Any]) -> Dict[str, Any]:
        symbols = index.get("symbols", [])
        if not isinstance(symbols, list):
            return {}

        # Build called_by map
        called_by_map = defaultdict(list)
        for item in symbols:
            if not isinstance(item, dict):
                continue
            caller = item.get("name")
            for callee in item.get("calls", []):
                if caller not in called_by_map[callee]:
                    called_by_map[callee].append(caller)

        result = {}
        for item in symbols:
            if not isinstance(item, dict):
                continue
            name = item.get("name", "unknown")
            result[name] = {
                "type": item.get("type"),
                "file": item.get("file"),
                "calls": item.get("calls", []),
                "called_by": called_by_map.get(name, [])
            }

        return result

    def _extract_dependencies(self, graph: Dict[str, Any]) -> Dict[str, List[str]]:
        return graph.get("calls", {})

    def _find_hotspots(self, graph: Dict[str, Any]) -> List[Dict[str, Any]]:
        calls = graph.get("calls", {})
        called_by = graph.get("called_by", {})

        hotspots = []

        for node, edges in calls.items():
            fan_out_count = len(edges)
            if fan_out_count > 5:
                hotspots.append({
                    "symbol": node,
                    "type": "fan_out",
                    "metric_type": "high_coupling",
                    "count": fan_out_count,
                    "description": f"Calls {fan_out_count} other functions"
                })

        for node, callers in called_by.items():
            fan_in_count = len(callers)
            if fan_in_count > 5:
                hotspots.append({
                    "symbol": node,
                    "type": "fan_in",
                    "metric_type": "high_usage",
                    "count": fan_in_count,
                    "description": f"Called by {fan_in_count} other functions"
                })

        return sorted(hotspots, key=lambda x: x["count"], reverse=True)

    def _build_arch_summary(self, graph: Dict[str, Any]) -> Dict[str, Any]:
        calls = graph.get("calls", {})
        called_by = graph.get("called_by", {})

        total_nodes = len(calls)
        total_edges = sum(len(v) for v in calls.values())
        all_nodes = set(calls.keys()) | set(called_by.keys())
        connected_nodes = set(calls.keys()) | set(called_by.keys())
        isolated = len(all_nodes) - len(connected_nodes)

        return {
            "total_nodes": total_nodes,
            "total_edges": total_edges,
            "avg_fan_out": total_edges / max(total_nodes, 1),
            "max_fan_out": max((len(v) for v in calls.values()), default=0),
            "max_fan_in": max((len(v) for v in called_by.values()), default=0),
            "isolated_functions": isolated
        }

import json
from pathlib import Path
from collections import defaultdict, deque


class ImpactAnalysisService:
    def __init__(self, index_path: str):
        self.index_path = Path(index_path)
        self.symbols = self._load_symbols()
        self.graph = self._build_reverse_graph()

    def _load_symbols(self):
        file = self.index_path / "symbols.json"

        if not file.exists():
            return []

        return json.loads(file.read_text(encoding="utf-8"))

    def _build_reverse_graph(self):
        reverse_graph = defaultdict(list)

        for sym in self.symbols:
            caller = sym["name"]
            calls = sym.get("calls", [])

            for callee in calls:
                reverse_graph[callee].append(caller)

        return reverse_graph

    def analyze(self, symbol_name: str, max_depth: int = 5):
        """
        Returns all impacted functions (who depends on this symbol)
        """

        visited = set()
        queue = deque([(symbol_name, 0)])
        impacted = set()

        while queue:
            current, depth = queue.popleft()

            if depth >= max_depth:
                continue

            for caller in self.graph.get(current, []):
                if caller not in visited:
                    visited.add(caller)
                    impacted.add(caller)
                    queue.append((caller, depth + 1))

        return {
            "symbol": symbol_name,
            "max_depth": max_depth,
            "impacted_count": len(impacted),
            "impacted_symbols": list(impacted)
        }

import time


class ContextCache:
    """
    Simple in-memory cache for symbol context + reasoning.
    """

    def __init__(self, ttl_seconds: int = 300):
        self.ttl = ttl_seconds
        self.cache = {}

    def get(self, key: str):
        entry = self.cache.get(key)

        if not entry:
            return None

        value, timestamp = entry

        if time.time() - timestamp > self.ttl:
            del self.cache[key]
            return None

        return value

    def set(self, key: str, value):
        self.cache[key] = (value, time.time())

    def clear(self):
        self.cache.clear()

import json
from pathlib import Path
from typing import Dict, List, Any, Optional


class ContextBuilder:
    """
    Unified context builder for symbol analysis.
    Loads from index files and provides rich symbol context.
    """

    def __init__(self, index_path: Path):
        self.index_path = Path(index_path)
        self.symbols = self._load_symbols()
        self.name_index = {s["name"]: s for s in self.symbols}
        self._callers_cache = None

    def _load_symbols(self) -> List[Dict[str, Any]]:
        """Load symbols from index file"""
        file = self.index_path / "symbols.json"
        if not file.exists():
            return []
        return json.loads(file.read_text(encoding="utf-8"))

    def _build_callers_map(self) -> Dict[str, List[str]]:
        """Build reverse mapping of callers for each symbol"""
        if self._callers_cache is not None:
            return self._callers_cache

        callers_map = {}
        for s in self.symbols:
            caller = s["name"]
            for callee in s.get("calls", []):
                if callee not in callers_map:
                    callers_map[callee] = []
                callers_map[callee].append(caller)

        self._callers_cache = callers_map
        return callers_map

    def build(self, symbol_name: str, intent: str = None) -> Optional[Dict[str, Any]]:
        """
        Build context for a symbol based on intent.
        
        Args:
            symbol_name: Name of the symbol to analyze
            intent: Optional intent to customize output (explain_symbol, find_callers, impact_analysis, dependency_analysis)
        
        Returns:
            Dictionary with symbol context, or None if symbol not found
        """
        symbol = self.name_index.get(symbol_name)
        if not symbol:
            return None

        if intent == "find_callers":
            return self._build_find_callers(symbol_name)

        if intent == "impact_analysis":
            return self._build_impact(symbol)

        if intent == "dependency_analysis":
            return self._build_dependency(symbol)

        # Default to explain_symbol
        return self._build_explain(symbol)

    # -----------------------------------
    # BUILD METHODS
    # -----------------------------------

    def _build_explain(self, symbol: Dict[str, Any]) -> Dict[str, Any]:
        """Build context for symbol explanation"""
        return {
            "symbol": symbol["name"],
            "type": symbol.get("type"),
            "parent": symbol.get("parent"),
            "file": symbol.get("file"),
            "calls": symbol.get("calls", []),
            "called_by": self._find_callers(symbol["name"]),
            "related_symbols": self._find_related(symbol)
        }

    def _build_find_callers(self, symbol_name: str) -> Dict[str, Any]:
        """Build context focused on callers"""
        return {
            "symbol": symbol_name,
            "called_by": self._find_callers(symbol_name)
        }

    def _build_impact(self, symbol: Dict[str, Any]) -> Dict[str, Any]:
        """Build context for impact analysis"""
        return {
            "symbol": symbol["name"],
            "type": symbol.get("type"),
            "parent": symbol.get("parent"),
            "file": symbol.get("file"),
            "calls": symbol.get("calls", []),
            "called_by": self._find_callers(symbol["name"]),
            "related_symbols": self._find_related(symbol)
        }

    def _build_dependency(self, symbol: Dict[str, Any]) -> Dict[str, Any]:
        """Build context for dependency analysis"""
        return {
            "symbol": symbol["name"],
            "file": symbol.get("file"),
            "parent": symbol.get("parent"),
            "calls": symbol.get("calls", [])
        }

    # -----------------------------------
    # HELPER METHODS
    # -----------------------------------

    def _find_callers(self, symbol_name: str) -> List[Dict[str, Any]]:
        """Find all symbols that call the given symbol"""
        callers_map = self._build_callers_map()
        caller_names = callers_map.get(symbol_name, [])

        callers = []
        for name in caller_names:
            s = self.name_index.get(name)
            if s:
                callers.append({
                    "name": s["name"],
                    "type": s.get("type"),
                    "parent": s.get("parent")
                })

        return callers

    def _find_related(self, symbol: Dict[str, Any]) -> List[str]:
        """Find symbols related by parent class/scope"""
        related = []
        parent = symbol.get("parent")

        if parent:
            for s in self.symbols:
                if s.get("parent") == parent and s["name"] != symbol["name"]:
                    related.append(f"{parent}.{s['name']}")

        return related

    def get_symbol_source_location(self, symbol_name: str) -> Optional[Dict[str, Any]]:
        """Get file and line information for a symbol"""
        symbol = self.name_index.get(symbol_name)
        if not symbol:
            return None
        return {
            "file": symbol.get("file"),
            "start_line": symbol.get("start_line"),
            "end_line": symbol.get("end_line")
        }

import json
from pathlib import Path
from collections import defaultdict
from typing import Dict, List, Any, Optional


class CallGraphService:
    """Service for building and querying call graphs from indexed symbols"""
    
    def __init__(self, index_path: str):
        self.index_path = Path(index_path)
        self._symbols_cache = None
        self._graph_cache = None

    def _load_symbols(self) -> List[Dict[str, Any]]:
        """Load symbols from index file with caching"""
        if self._symbols_cache is not None:
            return self._symbols_cache

        file = self.index_path / "symbols.json"
        if not file.exists():
            print(f"[GraphService] symbols.json not found at: {file}")
            self._symbols_cache = []
            return self._symbols_cache

        self._symbols_cache = json.loads(file.read_text(encoding="utf-8"))
        return self._symbols_cache

    def build_graph(self) -> Dict[str, Any]:
        """
        Build call graph from symbols.
        Results are cached for performance.
        """
        if self._graph_cache is not None:
            return self._graph_cache

        symbols = self._load_symbols()
        graph = defaultdict(list)
        reverse_graph = defaultdict(list)

        for sym in symbols:
            caller = sym.get("name")
            calls = sym.get("calls", [])

            for callee in calls:
                graph[caller].append(callee)
                reverse_graph[callee].append(caller)

        self._graph_cache = {
            "calls": dict(graph),
            "called_by": dict(reverse_graph)
        }
        return self._graph_cache

    def invalidate_cache(self) -> None:
        """Invalidate cached graph (call after reindexing)"""
        self._graph_cache = None
        self._symbols_cache = None

    def get_dependencies(self, function_name: str) -> List[str]:
        """Get functions that this function calls"""
        graph = self.build_graph()
        return graph.get("calls", {}).get(function_name, [])

    def get_dependents(self, function_name: str) -> List[str]:
        """Get functions that call this function"""
        graph = self.build_graph()
        return graph.get("called_by", {}).get(function_name, [])

    def get_call_chain(self, function_name: str, direction: str = "both", 
                       max_depth: int = 3) -> Dict[str, Any]:
        """
        Get call chain up to max_depth.
        
        Args:
            function_name: Starting function
            direction: 'up' (callers), 'down' (callees), or 'both'
            max_depth: Maximum depth to traverse
        """
        graph = self.build_graph()
        calls = graph.get("calls", {})
        called_by = graph.get("called_by", {})

        result = {
            "root": function_name,
            "callers": [],
            "callees": []
        }

        if direction in ("up", "both"):
            result["callers"] = self._traverse_up(function_name, called_by, max_depth)

        if direction in ("down", "both"):
            result["callees"] = self._traverse_down(function_name, calls, max_depth)

        return result

    def _traverse_up(self, node: str, called_by: Dict, max_depth: int) -> List[Dict]:
        """Traverse up the call chain (find callers)"""
        result = []
        visited = set()
        queue = [(node, 0)]

        while queue:
            current, depth = queue.pop(0)
            if depth >= max_depth or current in visited:
                continue
            visited.add(current)

            for caller in called_by.get(current, []):
                result.append({
                    "from": caller,
                    "to": current,
                    "depth": depth + 1
                })
                queue.append((caller, depth + 1))

        return result

    def _traverse_down(self, node: str, calls: Dict, max_depth: int) -> List[Dict]:
        """Traverse down the call chain (find callees)"""
        result = []
        visited = set()
        queue = [(node, 0)]

        while queue:
            current, depth = queue.pop(0)
            if depth >= max_depth or current in visited:
                continue
            visited.add(current)

            for callee in calls.get(current, []):
                result.append({
                    "from": current,
                    "to": callee,
                    "depth": depth + 1
                })
                queue.append((callee, depth + 1))

        return result

import ast
from .utils import generate_id


class CallVisitor(ast.NodeVisitor):
    def __init__(self):
        self.calls = []

    def visit_Call(self, node):
        if isinstance(node.func, ast.Name):
            self.calls.append(node.func.id)
        elif isinstance(node.func, ast.Attribute):
            self.calls.append(node.func.attr)

        self.generic_visit(node)


class ImportVisitor(ast.NodeVisitor):
    def __init__(self):
        self.imports = []

    def visit_Import(self, node):
        for alias in node.names:
            self.imports.append({
                "module": alias.name,
                "alias": alias.asname,
                "type": "unknown"  # Could be classified later
            })

    def visit_ImportFrom(self, node):
        module = node.module or ""
        for alias in node.names:
            self.imports.append({
                "module": f"{module}.{alias.name}" if module else alias.name,
                "alias": alias.asname,
                "type": "unknown"
            })


class PythonParser:

    def parse(self, file_path: str, source_code: str):
        try:
            tree = ast.parse(source_code)
        except SyntaxError as e:
            print(f"Syntax error in {file_path}: {e}")
            return [], []

        symbols = []
        imports = []

        # Parse imports first
        import_visitor = ImportVisitor()
        import_visitor.visit(tree)
        imports = import_visitor.imports

        for node in tree.body:
            # ---------------- FUNCTION ----------------
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                visitor = CallVisitor()
                visitor.visit(node)

                symbols.append({
                    "symbol_id": generate_id(),
                    "name": node.name,
                    "type": "function",
                    "file": file_path,
                    "start_line": node.lineno,
                    "end_line": getattr(node, "end_lineno", node.lineno),
                    "parent": None,
                    "calls": visitor.calls,
                    "referenced_vars": []
                })

            # ---------------- CLASS ----------------
            elif isinstance(node, ast.ClassDef):
                # Class itself (no calls)
                symbols.append({
                    "symbol_id": generate_id(),
                    "name": node.name,
                    "type": "class",
                    "file": file_path,
                    "start_line": node.lineno,
                    "end_line": getattr(node, "end_lineno", node.lineno),
                    "parent": None,
                    "calls": [],
                    "referenced_vars": []
                })

                # ---------------- METHODS ----------------
                for child in node.body:
                    if isinstance(child, (ast.FunctionDef, ast.AsyncFunctionDef)):
                        method_visitor = CallVisitor()
                        method_visitor.visit(child)

                        symbols.append({
                            "symbol_id": generate_id(),
                            "name": child.name,
                            "type": "method",
                            "file": file_path,
                            "start_line": child.lineno,
                            "end_line": getattr(child, "end_lineno", child.lineno),
                            "parent": node.name,
                            "calls": method_visitor.calls,
                            "referenced_vars": []
                        })

        return symbols, imports

from pathlib import Path
from typing import List


class ProjectScanner:
    def __init__(self, root_path: str, ignore_dirs=None):
        self.root_path = Path(root_path)
        self.ignore_dirs = ignore_dirs or {".git", "__pycache__", "venv", ".venv"}

    def scan(self) -> List[Path]:
        files = []

        for path in self.root_path.rglob("*"):
            if path.is_dir():
                continue

            if any(part in self.ignore_dirs for part in path.parts):
                continue

            if path.suffix == ".py":
                files.append(path)

        return files

class IndexSerializer:
    """Handles saving and loading index data to/from disk"""

        project_dir = self._ensure_dir(project_id)

        files = index_data.get("files", [])
        symbols = index_data.get("symbols", [])
        imports = index_data.get("imports", [])

        # Save individual files
        (project_dir / "files.json").write_text(
            json.dumps(files, indent=2, default=str), 
            encoding="utf-8"
        )
        (project_dir / "symbols.json").write_text(
            json.dumps(symbols, indent=2, default=str), 
            encoding="utf-8"
        )
        (project_dir / "imports.json").write_text(
            json.dumps(imports, indent=2, default=str), 
            encoding="utf-8"
        )

        # Create manifest
        manifest = {
            "project_id": project_id,
            "files_count": len(files),
            "symbols_count": len(symbols),
            "imports_count": len(imports),
            "version": "1.0"
        }

        (project_dir / "manifest.json").write_text(
            json.dumps(manifest, indent=2), 
            encoding="utf-8"
        )

        return manifest

    def load(self, project_id: str) -> Dict[str, Any]:
        """
        Load index data from disk.
        
        Returns:
            Dictionary with 'files', 'symbols', 'imports' keys
        """
        project_dir = self.base_path / project_id
        
        files = []
        symbols = []
        imports = []
        
        files_path = project_dir / "files.json"
        if files_path.exists():
            files = json.loads(files_path.read_text(encoding="utf-8"))
        
        symbols_path = project_dir / "symbols.json"
        if symbols_path.exists():
            symbols = json.loads(symbols_path.read_text(encoding="utf-8"))
        
        imports_path = project_dir / "imports.json"
        if imports_path.exists():
            imports = json.loads(imports_path.read_text(encoding="utf-8"))
        
        return {
            "files": files,
            "symbols": symbols,
            "imports": imports
        }
    
    def manifest_exists(self, project_id: str) -> bool:
        """Check if a project has been indexed"""
        manifest_path = self.base_path / project_id / "manifest.json"
        return manifest_path.exists()

class OllamaClient:

        self,
        base_url="http://localhost:11434",

    ):
        self.base_url = base_url
        self.model = model

    def generate(self, prompt: str):
        url = f"{self.base_url}/api/generate"
        payload = {
            "model": self.model,
            "prompt": prompt,
            "stream": False,
            "options": {
                "temperature": 0.1,
                "num_predict": 2048
            }
        }

        start = time.perf_counter()
        response = requests.post(url, json=payload)
        elapsed = time.perf_counter() - start

        print("=" * 60)
        print(f"Ollama inference: {elapsed:.2f}s")
        print("=" * 60)

        if response.status_code != 200:
            return {
                "error": "LLM request failed",
                "details": response.text
            }

        data = response.json()
        return data.get("response", "")

    def generate_stream(self, prompt: str) -> Generator[str, None, None]:
        """
        Streams response tokens dynamically from the local Ollama instance line-by-line.
        """
        url = f"{self.base_url}/api/generate"
        payload = {
            "model": self.model,
            "prompt": prompt,
            "stream": True,
            "options": {
                "temperature": 0.1
            }
        }

        try:
            response = requests.post(url, json=payload, stream=True)
            if response.status_code != 200:
                yield f"[ERROR: Ollama server returned status {response.status_code}]"
                return

            for line in response.iter_lines():
                if line:
                    chunk = json.loads(line.decode("utf-8"))
                    token = chunk.get("response", "")
                    yield token
                    if chunk.get("done", False):
                        break
        except requests.exceptions.RequestException as e:
            yield f"[ERROR: Failed to reach Ollama instance. Details: {e}]"

import ast
import re
from pathlib import Path
from typing import List, Dict, Any, Optional, Tuple
from dataclasses import dataclass
import uuid

from backend.models.refactor import (
    RefactorType, Parameter, CallSite, FunctionDetail,
    FileChange, ChangeType, RefactoringPlan
)
from backend.services.refactor.code_analyzer import CodeAnalyzer
from backend.services.refactor.impact_scanner import ImpactScanner


class ChangeEngine:
    """
    Generates concrete code changes for refactoring operations.
    Handles both simple (AST-based) and complex (LLM-based) changes.
    """
    
    def __init__(self, project_id: str = None):
        self.analyzer = CodeAnalyzer(project_id)
        self.scanner = ImpactScanner(project_id)
        self.project_id = project_id
    
    def generate_add_parameter(
        self,
        function_name: str,
        new_param: Parameter,
        default_value: Optional[str] = None,
        position: Optional[int] = None
    ) -> RefactoringPlan:
        """
        Generate changes for adding a parameter to a function.
        
        Args:
            function_name: Name of target function
            new_param: Parameter to add
            default_value: Default value for parameter
            position: Position to insert (None = end)
        """
        func_detail = self.analyzer.get_function_detail(function_name)
        if not func_detail:
            raise ValueError(f"Function '{function_name}' not found")
        
        changes = []
        
        # Set default value if provided
        if default_value:
            new_param.default_value = default_value
            new_param.is_required = False
        
        # 1. Update function definition
        func_change = self._update_function_definition(
            func_detail, new_param, position
        )
        changes.append(func_change)
        
        # 2. Update all callers
        for caller in func_detail.callers:
            caller_change = self._update_caller_for_added_param(
                caller, function_name, new_param, default_value
            )
            if caller_change:
                changes.append(caller_change)
        
        # Calculate confidence (simple changes = high confidence)
        confidence = 0.9 if len(func_detail.callers) <= 5 else 0.7
        
        return RefactoringPlan(
            id=str(uuid.uuid4()),
            target_function=function_name,
            refactor_type=RefactorType.ADD_PARAMETER,
            changes=changes,
            affected_files=list(set([c.file_path for c in changes])),
            confidence=confidence,
            requires_review=len(func_detail.callers) > 10,
            warning=None if len(func_detail.callers) <= 10 else "Many callers to update - review recommended"
        )
    
    def generate_remove_parameter(
        self,
        function_name: str,
        parameter_name: str
    ) -> RefactoringPlan:
        """
        Generate changes for removing a parameter from a function.
        
        Args:
            function_name: Name of target function
            parameter_name: Name of parameter to remove
        """
        func_detail = self.analyzer.get_function_detail(function_name)
        if not func_detail:
            raise ValueError(f"Function '{function_name}' not found")
        
        # Find parameter
        target_param = None
        param_index = -1
        for i, p in enumerate(func_detail.parameters):
            if p.name == parameter_name:
                target_param = p
                param_index = i
                break
        
        if not target_param:
            raise ValueError(f"Parameter '{parameter_name}' not found")
        
        changes = []
        
        # 1. Update function definition
        func_change = self._update_function_definition_remove(
            func_detail, param_index, parameter_name
        )
        changes.append(func_change)
        
        # 2. Update callers that use this parameter
        for caller in func_detail.callers:
            if self._caller_uses_parameter(caller, param_index, parameter_name):
                caller_change = self._update_caller_for_removed_param(
                    caller, function_name, param_index, parameter_name
                )
                if caller_change:
                    changes.append(caller_change)
        
        confidence = 0.85 if len(func_detail.callers) <= 5 else 0.7
        
        return RefactoringPlan(
            id=str(uuid.uuid4()),
            target_function=function_name,
            refactor_type=RefactorType.REMOVE_PARAMETER,
            changes=changes,
            affected_files=list(set([c.file_path for c in changes])),
            confidence=confidence,
            requires_review=True,  # Removing parameters always requires review
            warning="Removing parameters can break client code. Test thoroughly."
        )
    
    def generate_rename_function(
        self,
        function_name: str,
        new_name: str
    ) -> RefactoringPlan:
        """
        Generate changes for renaming a function.
        
        Args:
            function_name: Current function name
            new_name: New function name
        """
        func_detail = self.analyzer.get_function_detail(function_name)
        if not func_detail:
            raise ValueError(f"Function '{function_name}' not found")
        
        changes = []
        
        # 1. Update function definition
        func_change = self._rename_function_definition(
            func_detail, new_name
        )
        changes.append(func_change)
        
        # 2. Update all callers
        for caller in func_detail.callers:
            caller_change = self._rename_function_in_caller(
                caller, function_name, new_name
            )
            if caller_change:
                changes.append(caller_change)
        
        confidence = 0.95  # Rename is straightforward
        
        return RefactoringPlan(
            id=str(uuid.uuid4()),
            target_function=function_name,
            refactor_type=RefactorType.RENAME_FUNCTION,
            changes=changes,
            affected_files=list(set([c.file_path for c in changes])),
            confidence=confidence,
            requires_review=len(func_detail.callers) > 5,
            warning="Function renaming may affect dynamic imports or reflection" if len(func_detail.callers) > 10 else None
        )
    
    def _update_function_definition(
        self,
        func_detail: FunctionDetail,
        new_param: Parameter,
        position: Optional[int] = None
    ) -> FileChange:
        """Update function definition to include new parameter"""
        
        lines = func_detail.source_code.split('\n')
        if not lines:
            raise ValueError("Empty source code")
        
        # Find the function definition line
        def_line_index = 0
        for i, line in enumerate(lines):
            if re.match(rf'(async\s+)?def\s+{re.escape(func_detail.name)}\s*\(', line):
                def_line_index = i
                break
        
        def_line = lines[def_line_index]
        
        # Build new parameter string
        param_str = new_param.name
        if new_param.type_hint:
            param_str += f": {new_param.type_hint}"
        if new_param.default_value:
            param_str += f" = {new_param.default_value}"
        
        # Insert parameter
        if position is not None and position < len(func_detail.parameters):
            # Insert at specific position
            # This is complex; for simplicity, always add to end
            pass
        
        # Add to end of parameters
        if '(' in def_line and ')' in def_line:
            # Find where to insert
            if def_line.rstrip().endswith('):'):
                # No parameters currently
                new_def_line = def_line.replace('):', f', {param_str}):')
            else:
                # Has existing parameters
                new_def_line = def_line.replace(')', f', {param_str})')
        else:
            new_def_line = def_line  # Fallback
        
        before = def_line
        after = new_def_line
        
        return FileChange(
            file_path=func_detail.file_path,
            line_start=func_detail.line_start + def_line_index,
            line_end=func_detail.line_start + def_line_index,
            before=before,
            after=after,
            change_type=ChangeType.FUNCTION_DEFINITION,
            description=f"Add parameter '{new_param.name}' to function '{func_detail.name}'"
        )
    
    def _update_caller_for_added_param(
        self,
        caller: CallSite,
        function_name: str,
        new_param: Parameter,
        default_value: Optional[str]
    ) -> Optional[FileChange]:
        """Update a caller to include the new parameter"""
        
        if not default_value and new_param.default_value:
            default_value = new_param.default_value
        
        # If no default value, need to add argument
        if not default_value:
            # Check if caller already has the argument (shouldn't happen)
            if any(function_name in arg for arg in caller.arguments):
                return None
            
            # Add argument with default value or None
            new_arg = f"{new_param.name}={default_value or 'None'}"
            
            # Update the call line
            old_line = caller.source_line
            if '(' in old_line and ')' in old_line:
                if old_line.rstrip().endswith('):'):
                    new_line = old_line.replace(')', f', {new_arg})')
                elif old_line.rstrip().endswith(')'):
                    new_line = old_line.replace(')', f', {new_arg})')
                else:
                    new_line = old_line
            else:
                new_line = old_line
            
            return FileChange(
                file_path=caller.file_path,
                line_start=caller.line_number,
                line_end=caller.line_number,
                before=old_line,
                after=new_line,
                change_type=ChangeType.CALLER_UPDATE,
                description=f"Add '{new_arg}' argument when calling '{function_name}'"
            )
        
        return None  # No change needed if parameter has default
    
    def _update_function_definition_remove(
        self,
        func_detail: FunctionDetail,
        param_index: int,
        param_name: str
    ) -> FileChange:
        """Update function definition to remove a parameter"""
        
        lines = func_detail.source_code.split('\n')
        if not lines:
            raise ValueError("Empty source code")
        
        # Find the function definition line
        def_line_index = 0
        for i, line in enumerate(lines):
            if re.match(rf'(async\s+)?def\s+{re.escape(func_detail.name)}\s*\(', line):
                def_line_index = i
                break
        
        def_line = lines[def_line_index]
        
        # Remove parameter from definition
        # Pattern: remove ", param_name" or "param_name, " or "param_name"
        patterns = [
            rf',\s*{re.escape(param_name)}(?:\s*:\s*[^=,]+)?(?:\s*=\s*[^,)]+)?',
            rf'{re.escape(param_name)}(?:\s*:\s*[^=,]+)?(?:\s*=\s*[^,)]+)?,\s*',
            rf'{re.escape(param_name)}(?:\s*:\s*[^=,]+)?(?:\s*=\s*[^,)]+)?'
        ]
        
        new_def_line = def_line
        for pattern in patterns:
            new_def_line = re.sub(pattern, '', new_def_line)
            if new_def_line != def_line:
                break
        
        # Clean up double commas
        new_def_line = re.sub(r',\s*,', ',', new_def_line)
        # Clean up spacing
        new_def_line = re.sub(r'\(\s*,', '(', new_def_line)
        new_def_line = re.sub(r',\s*\)', ')', new_def_line)
        
        return FileChange(
            file_path=func_detail.file_path,
            line_start=func_detail.line_start + def_line_index,
            line_end=func_detail.line_start + def_line_index,
            before=def_line,
            after=new_def_line,
            change_type=ChangeType.FUNCTION_DEFINITION,
            description=f"Remove parameter '{param_name}' from function '{func_detail.name}'"
        )
    
    def _update_caller_for_removed_param(
        self,
        caller: CallSite,
        function_name: str,
        param_index: int,
        param_name: str
    ) -> Optional[FileChange]:
        """Update a caller to remove the argument for a deleted parameter"""
        
        old_line = caller.source_line
        
        # Try to remove the argument
        # Patterns: param_name=value, or just value if positional
        patterns = [
            rf'{re.escape(param_name)}\s*=\s*[^,)]+,\s*',
            rf',\s*{re.escape(param_name)}\s*=\s*[^,)]+',
            rf'{re.escape(param_name)}\s*=\s*[^,)]+'
        ]
        
        new_line = old_line
        for pattern in patterns:
            new_line = re.sub(pattern, '', new_line)
            if new_line != old_line:
                break
        
        # If still same, try positional removal
        if new_line == old_line and param_index < len(caller.arguments):
            # Remove by position (simplified)
            args = caller.arguments
            if param_index < len(args):
                del args[param_index]
                # Rebuild call line (simplified)
                call_start = re.search(rf'{re.escape(function_name)}\s*\(', new_line)
                if call_start:
                    new_line = f"{function_name}({', '.join(args)})"
        
        if new_line != old_line:
            return FileChange(
                file_path=caller.file_path,
                line_start=caller.line_number,
                line_end=caller.line_number,
                before=old_line,
                after=new_line,
                change_type=ChangeType.CALLER_UPDATE,
                description=f"Remove argument for '{param_name}' when calling '{function_name}'"
            )
        
        return None
    
    def _rename_function_definition(
        self,
        func_detail: FunctionDetail,
        new_name: str
    ) -> FileChange:
        """Rename function in its definition"""
        
        lines = func_detail.source_code.split('\n')
        def_line_index = 0
        for i, line in enumerate(lines):
            if re.match(rf'(async\s+)?def\s+{re.escape(func_detail.name)}\s*\(', line):
                def_line_index = i
                break
        
        def_line = lines[def_line_index]
        new_def_line = re.sub(
            rf'(async\s+)?def\s+{re.escape(func_detail.name)}',
            rf'\g<1>def {new_name}',
            def_line
        )
        
        return FileChange(
            file_path=func_detail.file_path,
            line_start=func_detail.line_start + def_line_index,
            line_end=func_detail.line_start + def_line_index,
            before=def_line,
            after=new_def_line,
            change_type=ChangeType.FUNCTION_DEFINITION,
            description=f"Rename function '{func_detail.name}' to '{new_name}'"
        )
    
    def _rename_function_in_caller(
        self,
        caller: CallSite,
        old_name: str,
        new_name: str
    ) -> Optional[FileChange]:
        """Rename function call in caller"""
        
        old_line = caller.source_line
        new_line = re.sub(rf'\b{re.escape(old_name)}\b', new_name, old_line)
        
        if new_line != old_line:
            return FileChange(
                file_path=caller.file_path,
                line_start=caller.line_number,
                line_end=caller.line_number,
                before=old_line,
                after=new_line,
                change_type=ChangeType.CALLER_UPDATE,
                description=f"Update call to '{old_name}' -> '{new_name}'"
            )
        
        return None
    
    def _caller_uses_parameter(
        self,
        caller: CallSite,
        param_index: int,
        param_name: str
    ) -> bool:
        """Check if caller provides a value for the parameter"""
        # Check positional
        if param_index < caller.argument_count:
            return True
        # Check keyword
        if any(param_name in arg for arg in caller.arguments):
            return True
        return False
    
    def generate_plan(self, refactor_type: RefactorType, **kwargs) -> RefactoringPlan:
        """Generate a refactoring plan based on type"""
        if refactor_type == RefactorType.ADD_PARAMETER:
            return self.generate_add_parameter(
                kwargs['function_name'],
                kwargs['new_parameter'],
                kwargs.get('default_value'),
                kwargs.get('position')
            )
        elif refactor_type == RefactorType.REMOVE_PARAMETER:
            return self.generate_remove_parameter(
                kwargs['function_name'],
                kwargs['parameter_name']
            )
        elif refactor_type == RefactorType.RENAME_FUNCTION:
            return self.generate_rename_function(
                kwargs['function_name'],
                kwargs['new_name']
            )
        else:
            raise ValueError(f"Unsupported refactor type: {refactor_type}")

import ast
import re
from pathlib import Path
from typing import List, Optional, Dict, Any, Tuple
from dataclasses import dataclass

from backend.models.refactor import Parameter, CallSite, FunctionDetail
from backend.services.graph.call_graph_service import CallGraphService
from backend.services.project_service import ProjectService
from config import INDEX_DIR


class CodeAnalyzer:
    """Extracts detailed function information including call sites"""

    def __init__(self, project_id: str = None):
        if not project_id:
            project = ProjectService().get_active_project()
            if project:
                project_id = project["project_id"]
            else:
                raise ValueError("No active project found and no project_id provided")
        
        self.project_id = project_id
        self.index_path = INDEX_DIR / project_id
        self.graph_service = CallGraphService(str(self.index_path))
        
        # Cache for source code lines
        self._source_cache: Dict[str, List[str]] = {}
        self._symbols_cache: List[Dict] = None

    def get_function_detail(self, function_name: str) -> Optional[FunctionDetail]:
        """Get complete detail for a function including call sites"""
        
        # Get function info from graph service
        symbols = self._load_symbols()
        function_info = self._find_function_in_symbols(symbols, function_name)
        
        if not function_info:
            return None
        
        # Extract source code
        file_path = function_info.get("file")
        start_line = function_info.get("start_line")
        end_line = function_info.get("end_line")
        
        source_code = self._extract_source_code(file_path, start_line, end_line)
        
        # Parse parameters from source or AST
        parameters = self._extract_parameters(source_code, function_info)
        
        # Get callers - NOW USING CALL GRAPH SERVICE
        callers = self._get_callers_with_context(function_name)
        
        # Debug output
        print(f"[CodeAnalyzer] Function '{function_name}' has {len(callers)} caller(s)")
        
        return FunctionDetail(
            name=function_name,
            file_path=file_path,
            line_start=start_line,
            line_end=end_line,
            source_code=source_code,
            parameters=parameters,
            return_type=function_info.get("return_type"),
            decorators=function_info.get("decorators", []),
            is_async=function_info.get("is_async", False),
            is_method=bool(function_info.get("parent")),
            class_name=function_info.get("parent"),
            docstring=function_info.get("docstring"),
            callers=callers
        )

    def _load_symbols(self) -> List[Dict]:
        """Load symbols from index"""
        import json
        if self._symbols_cache is not None:
            return self._symbols_cache
            
        symbols_file = self.index_path / "symbols.json"
        if not symbols_file.exists():
            print(f"[CodeAnalyzer] symbols.json not found at {symbols_file}")
            self._symbols_cache = []
            return self._symbols_cache
        
        with open(symbols_file, 'r', encoding='utf-8') as f:
            self._symbols_cache = json.load(f)
        return self._symbols_cache

    def _find_function_in_symbols(self, symbols: List[Dict], function_name: str) -> Optional[Dict]:
        """Find function by name in symbols"""
        for sym in symbols:
            if sym.get("name") == function_name and sym.get("type") in ["function", "method"]:
                return sym
        return None

    def _extract_source_code(self, file_path: str, start_line: int, end_line: int) -> str:
        """Extract source code lines for a function"""
        if not file_path or not Path(file_path).exists():
            return ""
        
        # Cache file content
        if file_path not in self._source_cache:
            try:
                self._source_cache[file_path] = Path(file_path).read_text(encoding="utf-8").splitlines()
            except Exception as e:
                print(f"[CodeAnalyzer] Error reading {file_path}: {e}")
                return ""
        
        lines = self._source_cache[file_path]
        if start_line <= 0 or end_line > len(lines):
            return ""
        
        return "\n".join(lines[start_line - 1:end_line])

    def _extract_parameters(self, source_code: str, function_info: Dict) -> List[Parameter]:
        """Extract parameters from source code using AST"""
        parameters = []
        
        if not source_code or not source_code.strip():
            return parameters
        
        try:
            # Fix indentation issues
            import textwrap
            source_code = textwrap.dedent(source_code)
            
            tree = ast.parse(source_code)
            for node in ast.walk(tree):
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                    for arg in node.args.args:
                        # Skip 'self' and 'cls' for cleaner display
                        if arg.arg in ['self', 'cls']:
                            continue
                            
                        param = Parameter(
                            name=arg.arg,
                            type_hint=self._get_type_hint(arg),
                            default_value=None,
                            is_required=True
                        )
                        parameters.append(param)
                    
                    # Handle default values
                    defaults = node.args.defaults
                    if defaults:
                        offset = len(node.args.args) - len(defaults)
                        for i, default in enumerate(defaults):
                            if offset + i < len(parameters):
                                try:
                                    parameters[offset + i].default_value = ast.unparse(default)
                                    parameters[offset + i].is_required = False
                                except:
                                    parameters[offset + i].default_value = str(default)
                                    parameters[offset + i].is_required = False
                    break
        except SyntaxError as e:
            # Fallback: try to get parameters from function_info
            if 'parameters' in function_info:
                for param_info in function_info.get('parameters', []):
                    param = Parameter(
                        name=param_info.get('name', 'unknown'),
                        type_hint=param_info.get('type_hint'),
                        default_value=param_info.get('default_value'),
                        is_required=param_info.get('is_required', True)
                    )
                    parameters.append(param)
        except Exception as e:
            print(f"[CodeAnalyzer] Error parsing parameters: {e}")
        
        return parameters

    def _get_type_hint(self, arg: ast.arg) -> Optional[str]:
        """Extract type hint from argument"""
        if arg.annotation:
            try:
                return ast.unparse(arg.annotation)
            except:
                return None
        return None

    def _get_callers_with_context(self, function_name: str) -> List[CallSite]:
        """Get callers with source code context using call graph service"""
        call_sites = []
        
        # Use the call graph service to find callers
        graph = self.graph_service.build_graph()
        called_by = graph.get("called_by", {})
        caller_names = called_by.get(function_name, [])
        
        print(f"[CodeAnalyzer] Found {len(caller_names)} caller(s) in call graph for '{function_name}': {caller_names}")
        
        if not caller_names:
            return call_sites
        
        symbols = self._load_symbols()
        
        # Build name to symbol map for quick lookup
        symbol_map = {}
        for sym in symbols:
            name = sym.get("name")
            if name:
                symbol_map[name] = sym
        
        # Process each caller
        for caller_name in caller_names:
            caller_info = symbol_map.get(caller_name)
            if not caller_info:
                print(f"[CodeAnalyzer] Warning: Caller '{caller_name}' not found in symbols")
                continue
            
            file_path = caller_info.get("file")
            if not file_path:
                continue
            
            # Try to find the exact call site line
            call_lines = self._find_call_in_caller(file_path, caller_name, function_name)
            
            if call_lines:
                for line_num, source_line, args in call_lines:
                    call_sites.append(CallSite(
                        file_path=file_path,
                        line_number=line_num,
                        calling_function=caller_name,
                        arguments=args,
                        source_line=source_line.strip(),
                        argument_count=len(args)
                    ))
            else:
                # If we can't find exact line, create a basic call site
                call_sites.append(CallSite(
                    file_path=file_path,
                    line_number=caller_info.get("start_line", 0),
                    calling_function=caller_name,
                    arguments=[],
                    source_line=f"Call to {function_name} in {caller_name}",
                    argument_count=0
                ))
        
        return call_sites

    def _find_call_in_caller(self, file_path: str, caller_name: str, target_name: str) -> List[Tuple[int, str, List[str]]]:
        """Find the exact line where caller calls target function"""
        if not file_path or not Path(file_path).exists():
            return []
        
        # Cache file content
        if file_path not in self._source_cache:
            try:
                self._source_cache[file_path] = Path(file_path).read_text(encoding="utf-8").splitlines()
            except Exception as e:
                print(f"[CodeAnalyzer] Error reading {file_path}: {e}")
                return []
        
        lines = self._source_cache[file_path]
        
        # Find the caller function's range
        caller_start, caller_end = self._find_function_range(lines, caller_name)
        if caller_start < 0:
            return []
        
        results = []
        
        # Search within caller function for calls to target
        for i in range(caller_start - 1, min(caller_end, len(lines))):
            line = lines[i]
            # Look for pattern: target_name(...) or self.target_name(...) or obj.target_name(...)
            patterns = [
                rf'{re.escape(target_name)}\s*\(',  # direct call
                rf'self\.{re.escape(target_name)}\s*\(',  # method call
                rf'\w+\.{re.escape(target_name)}\s*\(',  # object method call
            ]
            
            for pattern in patterns:
                match = re.search(pattern, line)
                if match:
                    # Extract arguments
                    args_match = re.search(rf'{re.escape(target_name)}\s*\(([^)]*)\)', line)
                    args = []
                    if args_match:
                        args_str = args_match.group(1)
                        args = self._parse_arguments(args_str)
                    
                    results.append((i + 1, line, args))
                    break  # Found a match for this line
        
        return results

    def _find_function_range(self, lines: List[str], function_name: str) -> Tuple[int, int]:
        """Find start and end line numbers of a function"""
        start_line = -1
        indent_level = None
        
        for i, line in enumerate(lines):
            # Look for function definition (including methods with decorators)
            patterns = [
                rf'def\s+{re.escape(function_name)}\s*\(',
                rf'async\s+def\s+{re.escape(function_name)}\s*\(',
            ]
            
            for pattern in patterns:
                if re.search(pattern, line):
                    start_line = i + 1
                    indent_level = len(line) - len(line.lstrip())
                    break
            
            if start_line != -1:
                break
        
        if start_line < 0:
            return (-1, -1)
        
        # Find end of function (when indent returns to same or lower level)
        end_line = len(lines)
        for i in range(start_line, len(lines)):
            if i == start_line - 1:
                continue
            
            line = lines[i]
            if not line.strip():  # Skip empty lines
                continue
            
            if line.strip().startswith('#'):  # Skip comments
                continue
            
            current_indent = len(line) - len(line.lstrip())
            
            # Function ends when we hit a line with same or less indent
            # that starts a new definition or is not part of the function body
            if current_indent <= indent_level:
                # Check if this line starts a new function or class
                if re.match(r'\s*def\s+\w+\s*\(', line) or re.match(r'\s*class\s+\w+', line):
                    end_line = i
                    break
                # Also break if we're at module level (indent 0)
                elif current_indent == 0:
                    end_line = i
                    break
        
        return (start_line, end_line)

    def _parse_arguments(self, args_str: str) -> List[str]:
        """Parse arguments from a function call string"""
        if not args_str.strip():
            return []
        
        # Simple parsing - split by commas but respect nested parentheses and brackets
        args = []
        current = []
        paren_count = 0
        bracket_count = 0
        brace_count = 0
        
        for char in args_str:
            if char == '(':
                paren_count += 1
            elif char == ')':
                paren_count -= 1
            elif char == '[':
                bracket_count += 1
            elif char == ']':
                bracket_count -= 1
            elif char == '{':
                brace_count += 1
            elif char == '}':
                brace_count -= 1
            elif char == ',' and paren_count == 0 and bracket_count == 0 and brace_count == 0:
                args.append(''.join(current).strip())
                current = []
                continue
            current.append(char)
        
        if current:
            args.append(''.join(current).strip())
        
        # Clean up arguments (remove self for method calls)
        args = [arg for arg in args if arg and arg != 'self']
        
        return args

from pathlib import Path

class ImpactLevel(str, Enum):
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"
    CRITICAL = "critical"


@dataclass
class ImpactAnalysis:
    """Result of impact analysis for a refactoring operation"""
    impact_level: ImpactLevel
    affected_files: List[str]
    affected_callers: List[CallSite]
    total_callers: int
    breaking_changes: bool
    warnings: List[str]
    suggestions: List[str]
    complexity_score: float  # 0-1, higher means more complex
    
    def to_dict(self) -> Dict[str, Any]:
        return {
            "impact_level": self.impact_level.value,
            "affected_files": self.affected_files,
            "affected_callers": [c.to_dict() for c in self.affected_callers],
            "total_callers": self.total_callers,
            "breaking_changes": self.breaking_changes,
            "warnings": self.warnings,
            "suggestions": self.suggestions,
            "complexity_score": self.complexity_score
        }


class ImpactScanner:
    """
    Analyzes the impact of refactoring operations by examining
    call relationships and code complexity.
    """
    
    def __init__(self, project_id: str = None):
        self.analyzer = CodeAnalyzer(project_id)
    
    def analyze_add_parameter(
        self, 
        function_name: str, 
        new_parameter: Parameter,
        default_value: Optional[str] = None
    ) -> ImpactAnalysis:
        """
        Analyze impact of adding a new parameter to a function.
        
        Args:
            function_name: Target function name
            new_parameter: Parameter to add
            default_value: Default value (makes it non-breaking)
        """
        func_detail = self.analyzer.get_function_detail(function_name)
        if not func_detail:
            return self._error_analysis(f"Function '{function_name}' not found")
        
        warnings = []
        suggestions = []
        
        # Check if parameter already exists
        existing_params = {p.name for p in func_detail.parameters}
        if new_parameter.name in existing_params:
            return self._error_analysis(f"Parameter '{new_parameter.name}' already exists")
        
        # Determine if this is a breaking change
        has_default = default_value is not None or new_parameter.default_value is not None
        breaking_changes = not has_default and len(func_detail.callers) > 0
        
        # Assess impact based on callers
        total_callers = len(func_detail.callers)
        affected_files = list(set([c.file_path for c in func_detail.callers]))
        
        if breaking_changes:
            warnings.append(
                f"Adding required parameter will break {total_callers} caller(s). "
                f"Consider providing a default value."
            )
            suggestions.append(
                f"Add default value '{default_value or 'None'}' to make it non-breaking"
            )
        
        # Check call site complexity
        complex_callers = self._find_complex_callers(func_detail.callers)
        if complex_callers:
            warnings.append(
                f"{len(complex_callers)} caller(s) have complex argument expressions "
                f"that may need careful updating"
            )
        
        # Calculate impact level
        impact_level = self._calculate_impact_level(
            total_callers, 
            breaking_changes, 
            len(complex_callers),
            func_detail.is_public_api()
        )
        
        # Calculate complexity score
        complexity_score = self._calculate_complexity_score(
            total_callers,
            len(complex_callers),
            func_detail.parameters,
            has_default
        )
        
        return ImpactAnalysis(
            impact_level=impact_level,
            affected_files=affected_files,
            affected_callers=func_detail.callers,
            total_callers=total_callers,
            breaking_changes=breaking_changes,
            warnings=warnings,
            suggestions=suggestions,
            complexity_score=complexity_score
        )
    
    def analyze_remove_parameter(
        self,
        function_name: str,
        parameter_name: str
    ) -> ImpactAnalysis:
        """
        Analyze impact of removing a parameter from a function.
        
        Args:
            function_name: Target function name
            parameter_name: Name of parameter to remove
        """
        func_detail = self.analyzer.get_function_detail(function_name)
        if not func_detail:
            return self._error_analysis(f"Function '{function_name}' not found")
        
        warnings = []
        suggestions = []
        
        # Find the parameter
        target_param = None
        param_index = -1
        for i, p in enumerate(func_detail.parameters):
            if p.name == parameter_name:
                target_param = p
                param_index = i
                break
        
        if not target_param:
            return self._error_analysis(f"Parameter '{parameter_name}' not found")
        
        # Check if parameter is used in function body
        is_used = self._is_parameter_used(func_detail, parameter_name)
        if is_used:
            warnings.append(
                f"Parameter '{parameter_name}' appears to be used in the function body. "
                f"Removing it requires logic changes."
            )
        
        # Check if callers provide this argument
        callers_using_param = self._find_callers_using_parameter(
            func_detail.callers, 
            param_index,
            target_param
        )
        
        total_callers = len(callers_using_param)
        breaking_changes = total_callers > 0
        
        if breaking_changes:
            warnings.append(
                f"{total_callers} caller(s) provide this argument and will need to be updated"
            )
            suggestions.append(
                f"Consider deprecating the parameter first by adding a warning, "
                f"then remove in a future version"
            )
        
        affected_files = list(set([c.file_path for c in callers_using_param]))
        
        # Calculate impact level
        impact_level = self._calculate_impact_level(
            total_callers,
            breaking_changes,
            0,  # No complex callers for removal
            func_detail.is_public_api()
        )
        
        # Calculate complexity score
        complexity_score = self._calculate_complexity_score(
            total_callers,
            1 if is_used else 0,
            func_detail.parameters,
            False
        )
        
        return ImpactAnalysis(
            impact_level=impact_level,
            affected_files=affected_files,
            affected_callers=callers_using_param,
            total_callers=total_callers,
            breaking_changes=breaking_changes,
            warnings=warnings,
            suggestions=suggestions,
            complexity_score=complexity_score
        )
    
    def analyze_rename_function(
        self,
        function_name: str,
        new_name: str
    ) -> ImpactAnalysis:
        """
        Analyze impact of renaming a function.
        
        Args:
            function_name: Current function name
            new_name: Desired new name
        """
        func_detail = self.analyzer.get_function_detail(function_name)
        if not func_detail:
            return self._error_analysis(f"Function '{function_name}' not found")
        
        warnings = []
        suggestions = []
        
        total_callers = len(func_detail.callers)
        breaking_changes = total_callers > 0
        
        if breaking_changes:
            warnings.append(
                f"Renaming will break {total_callers} caller(s)"
            )
            suggestions.append(
                f"Keep old function as deprecated wrapper calling the new one"
            )
            suggestions.append(
                f"Use @deprecated decorator and schedule removal in future version"
            )
        
        # Check for name conflicts
        if self._function_exists(new_name):
            warnings.append(f"Function '{new_name}' already exists elsewhere")
            suggestions.append(f"Choose a different name")
        
        # Check if function is part of public API
        if func_detail.is_public_api():
            warnings.append(
                f"This appears to be a public API function. Renaming is a breaking change "

            )
        
        affected_files = list(set([c.file_path for c in func_detail.callers]))
        
        impact_level = self._calculate_impact_level(
            total_callers,
            breaking_changes,
            0,
            func_detail.is_public_api()
        )
        
        complexity_score = total_callers / 100.0 if total_callers < 100 else 1.0
        
        return ImpactAnalysis(
            impact_level=impact_level,
            affected_files=affected_files,
            affected_callers=func_detail.callers,
            total_callers=total_callers,
            breaking_changes=breaking_changes,
            warnings=warnings,
            suggestions=suggestions,
            complexity_score=complexity_score
        )
    
    def _find_complex_callers(self, callers: List[CallSite]) -> List[CallSite]:
        """Identify callers with complex argument expressions"""
        complex_patterns = [
            lambda x: '(' in x or ')' in x,  # Nested calls
            lambda x: '+' in x or '-' in x,   # Arithmetic
            lambda x: '[' in x or ']' in x,   # List/dict access
            lambda x: '.' in x,                # Attribute access
            lambda x: len(x) > 50,             # Very long expression
        ]
        
        complex_callers = []
        for caller in callers:
            for arg in caller.arguments:
                if any(pattern(arg) for pattern in complex_patterns):
                    complex_callers.append(caller)
                    break
        
        return complex_callers
    
    def _is_parameter_used(self, func_detail: FunctionDetail, param_name: str) -> bool:
        """Check if parameter is referenced in function body (simplified)"""
        # Simple check - look for parameter name in source code
        # In production, would use AST analysis
        if not func_detail.source_code:
            return False
        
        # Skip self/cls parameters
        if param_name in ['self', 'cls']:
            return False
        
        # Check if parameter appears after function definition
        lines = func_detail.source_code.split('\n')
        body_lines = lines[1:] if len(lines) > 1 else []
        
        for line in body_lines:
            # Look for param name as a word
            import re
            if re.search(rf'\b{re.escape(param_name)}\b', line):
                # Skip if in parameter list or docstring
                if not line.strip().startswith('"""') and not line.strip().startswith("'''"):
                    if 'def ' not in line:
                        return True
        
        return False
    
    def _find_callers_using_parameter(
        self, 
        callers: List[CallSite], 
        param_index: int,
        param: Parameter
    ) -> List[CallSite]:
        """Find callers that provide an argument for the specified parameter"""
        using_callers = []
        
        for caller in callers:
            # Positional argument
            if param_index < caller.argument_count:
                using_callers.append(caller)
            # Keyword argument
            elif any(param.name in arg for arg in caller.arguments):
                using_callers.append(caller)
        
        return using_callers
    
    def _function_exists(self, function_name: str) -> bool:
        """Check if a function with given name exists in the codebase"""
        # This would need to check symbols.json
        # Simplified for now
        return False
    
    def _calculate_impact_level(
        self,
        total_callers: int,
        breaking_changes: bool,
        complex_caller_count: int,
        is_public_api: bool
    ) -> ImpactLevel:
        """Calculate overall impact level"""
        if is_public_api and breaking_changes:
            return ImpactLevel.CRITICAL
        elif breaking_changes and total_callers > 20:
            return ImpactLevel.HIGH
        elif breaking_changes and total_callers > 5:
            return ImpactLevel.MEDIUM
        elif breaking_changes or complex_caller_count > 0:
            return ImpactLevel.MEDIUM
        elif total_callers == 0:
            return ImpactLevel.LOW
        else:
            return ImpactLevel.LOW
    
    def _calculate_complexity_score(
        self,
        total_callers: int,
        complex_caller_count: int,
        parameters: List[Parameter],
        has_default: bool
    ) -> float:
        """Calculate complexity score (0-1, higher is more complex)"""
        score = 0.0
        
        # Caller count factor
        if total_callers > 50:
            score += 0.4
        elif total_callers > 10:
            score += 0.2
        elif total_callers > 0:
            score += 0.1
        
        # Complex callers factor
        if complex_caller_count > 10:
            score += 0.3
        elif complex_caller_count > 0:
            score += 0.15
        
        # Parameter count factor
        if len(parameters) > 5:
            score += 0.2
        
        # Breaking vs non-breaking
        if not has_default and total_callers > 0:
            score += 0.2
        
        return min(score, 1.0)
    
    def _error_analysis(self, error_message: str) -> ImpactAnalysis:

        return ImpactAnalysis(

            affected_callers=[],
            total_callers=0,
            breaking_changes=True,
            warnings=[error_message],
            suggestions=["Verify function name and try again"],
            complexity_score=1.0
        )

import json
from pathlib import Path


class SymbolSearchService:
    def __init__(self, index_path: str):
        self.index_path = Path(index_path)
        self.symbols = self._load_symbols()

    def _load_symbols(self):
        symbols_file = self.index_path / "symbols.json"

        if not symbols_file.exists():
            return []

        return json.loads(symbols_file.read_text(encoding="utf-8"))

    def search(self, query: str):
        query = query.lower().strip()

        results = []

        for sym in self.symbols:
            name = sym.get("name", "").lower()
            sym_type = sym.get("type", "").lower()

            # MATCH RULES
            if query in name:
                results.append(sym)
                continue

            if query == sym_type:
                results.append(sym)
                continue

        return {
            "query": query,
            "count": len(results),
            "results": results
        }

    def search_by_type(self, symbol_type: str):
        symbol_type = symbol_type.lower()

        return [
            s for s in self.symbols
            if s.get("type", "").lower() == symbol_type
        ]

class BackgroundTaskManager:
    """Manages background task generation to prevent UI timeout"""
    
    _instance = None
    _jobs: Dict[str, Dict] = {}
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._jobs = {}
        return cls._instance
    
    def start_generation(self, project_id: str, include_llm: bool = True) -> str:
        """Start background task generation and return job ID"""
        
        job_id = str(uuid.uuid4())[:8]
        
        self._jobs[job_id] = {
            "id": job_id,
            "project_id": project_id,
            "include_llm": include_llm,
            "status": "running",
            "started_at": datetime.now().isoformat(),
            "completed_at": None,
            "result": None,
            "error": None
        }
        
        # Start background thread
        thread = threading.Thread(
            target=self._run_generation,
            args=(job_id, project_id, include_llm)
        )
        thread.daemon = True
        thread.start()
        
        return job_id
    
    def _run_generation(self, job_id: str, project_id: str, include_llm: bool):
        """Run task generation in background"""
        try:
            from datetime import datetime
            from backend.services.project_service import ProjectService
            from backend.models.task import TaskEvidence, AnalysisTask, TaskCategory, TaskPriority, TaskStatus
            
            # Get project config
            project = ProjectService().get_project(project_id)
            index_path = INDEX_DIR / project_id
            
            # Initialize services
            index_service = IndexService(project)
            graph_service = CallGraphService(index_path)
            crawler_service = CrawlerReviewService(index_service, graph_service)
            task_storage = TaskStorage(project_id, DATA_DIR)
            task_generator = TaskGenerator(crawler_service, graph_service, project_id)
            
            # Generate tasks
            new_tasks = task_generator.generate_tasks(include_llm=include_llm)
            
            # Load existing tasks
            existing_tasks = task_storage.load_tasks()
            existing_ids = {t["id"] for t in existing_tasks}
            
            # Add new tasks

class LLMTaskDiscovery:
    """Discovers tasks using LLM analysis of codebase"""
    
    def __init__(self, crawler_service, graph_service, config):
        self.crawler = crawler_service
        self.graph = graph_service
        self.config = config
        self.llm = OllamaClient(model=config.llm_model)
    
    def discover_tasks(self, max_tasks: int = None) -> List[AnalysisTask]:
        """Discover tasks by analyzing codebase with LLM"""
        
        if not self.config.llm_enabled:
            return []
        
        max_tasks = max_tasks or self.config.llm_max_tasks
        
        # Build context for LLM
        review_context = self.crawler.build_review_context()
        hotspots = review_context.get("hotspots", [])[:20]  # Top 20 hotspots
        arch_summary = review_context.get("architecture_summary", {})
        
        prompt = self._build_discovery_prompt(hotspots, arch_summary, max_tasks)
        
        try:
            response = self.llm.generate(prompt)
            tasks = self._parse_llm_response(response)
            return tasks[:max_tasks]
        except Exception as e:
            print(f"LLM task discovery failed: {e}")
            return []
    
    def _build_discovery_prompt(self, hotspots: List[Dict], arch_summary: Dict, max_tasks: int) -> str:
        """Build prompt for LLM to discover tasks"""
        
        hotspots_text = ""
        for h in hotspots[:10]:
            hotspots_text += f"- {h['symbol']}: {h['type']} = {h['count']}\n"
        
        return f"""You are a senior software architect analyzing a Python codebase.

ARCHITECTURE SUMMARY:
- Total functions: {arch_summary.get('total_nodes', 0)}
- Total dependencies: {arch_summary.get('total_edges', 0)}
- Average fan-out: {arch_summary.get('avg_fan_out', 0):.1f}
- Max fan-out: {arch_summary.get('max_fan_out', 0)}
- Max fan-in: {arch_summary.get('max_fan_in', 0)}

TOP HOTSPOTS (rule-based detection):
{hotspots_text}

Based on this analysis, suggest {max_tasks} specific, actionable improvements.

Focus on issues that rule-based detection might miss:
1. Design pattern violations or anti-patterns
2. Potential race conditions or async issues
3. Security vulnerabilities
4. Performance bottlenecks (N+1 queries, inefficient loops)
5. Error handling gaps
6. Code duplication opportunities
7. Wrong abstraction levels
8. Missing validation or sanitization
9. Thread safety concerns
10. Resource leaks (file handles, connections)

Return ONLY valid JSON in this exact format:
{{
  "tasks": [
    {{
      "title": "Brief title (max 60 chars)",
      "description": "Detailed explanation of the issue (2-3 sentences)",
      "category": "architecture|maintainability|performance|security|reliability",
      "priority": "critical|high|medium|low",
      "suggested_fix": "Specific code change recommendation",
      "symbol_hint": "Optional: specific function/class name if relevant"
    }}
  ]
}}

Do not include any text outside the JSON."""
    
    def _parse_llm_response(self, response: str) -> List[AnalysisTask]:
        """Parse LLM response into AnalysisTask objects"""
        
        tasks = []
        
        # Extract JSON from response
        json_match = re.search(r'\{[\s\S]*"tasks"[\s\S]*\}', response)
        if not json_match:
            print("No JSON found in LLM response")
            return []
        
        try:
            data = json.loads(json_match.group())
            raw_tasks = data.get("tasks", [])
            
            for raw in raw_tasks:
                # Map category string to TaskCategory enum
                category_str = raw.get("category", "maintainability")
                category_map = {
                    "architecture": TaskCategory.ARCHITECTURE,
                    "maintainability": TaskCategory.MAINTAINABILITY,
                    "performance": TaskCategory.PERFORMANCE,
                    "security": TaskCategory.SECURITY,
                    "reliability": TaskCategory.RELIABILITY,
                    "complexity": TaskCategory.COMPLEXITY
                }
                category = category_map.get(category_str, TaskCategory.MAINTAINABILITY)
                
                # Map priority string
                priority_str = raw.get("priority", "medium")
                priority_map = {
                    "critical": TaskPriority.CRITICAL,
                    "high": TaskPriority.HIGH,
                    "medium": TaskPriority.MEDIUM,
                    "low": TaskPriority.LOW
                }
                priority = priority_map.get(priority_str, TaskPriority.MEDIUM)
                
                # Create evidence with symbol hint if provided
                symbol_hint = raw.get("symbol_hint", "unknown")
                evidence = TaskEvidence(
                    symbol=symbol_hint,
                    metric_type="llm_discovered",
                    metric_value=0,
                    file_path="unknown",
                    source="llm"
                )
                
                task = AnalysisTask(
                    id=f"task_llm_{uuid.uuid4().hex[:12]}",
                    title=raw.get("title", "LLM Suggested Task")[:60],
                    description=raw.get("description", ""),
                    category=category,
                    priority=priority,
                    status=TaskStatus.PENDING,
                    evidence=[evidence],
                    suggested_fix=raw.get("suggested_fix", ""),
                    created_at=datetime.now(),
                    updated_at=datetime.now()
                )
                tasks.append(task)
                
        except json.JSONDecodeError as e:
            print(f"Failed to parse LLM response: {e}")
            print(f"Response excerpt: {response[:500]}")
        
        return tasks
    
    def generate_only_llm_tasks(self, max_tasks: int = None) -> List[AnalysisTask]:
        """
        Generate ONLY LLM-discovered tasks, no rule-based tasks.
        Useful for getting AI suggestions without rule-based noise.
        """
        if not self.config.llm_enabled:

        max_tasks = max_tasks or self.config.llm_max_tasks
        
        # Build lighter context for AI-only generation

        arch_summary = review_context.get("architecture_summary", {})
        
        prompt = self._build_ai_only_prompt(hotspots, arch_summary, max_tasks)

            tasks = self._parse_llm_response(response)
            return tasks[:max_tasks]
        except Exception as e:
            print(f"AI-only task discovery failed: {e}")
            return []

    def _build_ai_only_prompt(self, hotspots: List[Dict], arch_summary: Dict, max_tasks: int) -> str:

import uuid

class TaskGenerator:
    """Generates actionable tasks from codebase analysis"""

    def __init__(self, crawler_service: CrawlerReviewService, graph_service: CallGraphService, project_id: str):
        self.crawler = crawler_service
        self.graph = graph_service
        self.config = TaskConfig(project_id)
        self.llm_discovery = LLMTaskDiscovery(crawler_service, graph_service, self.config.get())

    def generate_tasks(self, include_llm: bool = True) -> List[AnalysisTask]:
        """Generate all tasks from current analysis"""
        review_context = self.crawler.build_review_context()
        hotspots = review_context.get("hotspots", [])
        symbols = review_context.get("symbols", {})

        tasks = []

        # Rule-based tasks from hotspots
        for hotspot in hotspots:
            task = self._create_task_from_hotspot(hotspot, symbols)
            if task:
                tasks.append(task)

        # LLM-discovered tasks
        if include_llm and self.config.get().llm_enabled:
            llm_tasks = self.llm_discovery.discover_tasks()
            tasks.extend(llm_tasks)

        return tasks

    def _create_task_from_hotspot(self, hotspot: Dict[str, Any], symbols: Dict) -> Optional[AnalysisTask]:
        """Create a task from a hotspot detection using configured thresholds"""
        hotspot_type = hotspot.get("type")
        symbol_name = hotspot.get("symbol")
        count = hotspot.get("count", 0)

        if hotspot_type == "fan_out":
            return self._create_coupling_task(symbol_name, count, symbols)
        elif hotspot_type == "fan_in":
            return self._create_usage_task(symbol_name, count, symbols)

        return None

    def _get_priority_from_threshold(self, count: int, medium: int, high: int, critical: int) -> TaskPriority:
        """Determine priority based on configured thresholds"""
        if count > critical:
            return TaskPriority.CRITICAL
        elif count > high:
            return TaskPriority.HIGH
        elif count > medium:
            return TaskPriority.MEDIUM
        else:
            return TaskPriority.LOW

    def _create_coupling_task(self, symbol_name: str, fan_out_count: int, symbols: Dict) -> AnalysisTask:
        """Create task for high coupling using configured thresholds"""
        symbol_info = symbols.get(symbol_name, {})
        
        priority = self._get_priority_from_threshold(
            fan_out_count,
            self.config.get().fan_out_medium,
            self.config.get().fan_out_high,
            self.config.get().fan_out_critical
        )

        evidence = TaskEvidence(
            symbol=symbol_name,
            metric_type="fan_out",
            metric_value=fan_out_count,
            file_path=symbol_info.get("file", "unknown"),
            callees=symbol_info.get("calls", [])[:20],
            source="rule_based"
        )

        suggested_fix = f"""Reduce coupling in {symbol_name} by extracting helper functions.

This function calls {fan_out_count} other functions. Move related calls into separate helper functions prefixed with '_'. Test each helper individually."""

        return AnalysisTask(
            id=f"task_coupling_{symbol_name}_{uuid.uuid4().hex[:8]}",
            title=f"High coupling in {symbol_name}",
            description=f"Function '{symbol_name}' calls {fan_out_count} other functions. This indicates too many responsibilities.",
            category=TaskCategory.HIGH_COUPLING,
            priority=priority,
            status=TaskStatus.PENDING,
            evidence=[evidence],
            suggested_fix=suggested_fix
        )

    def _create_usage_task(self, symbol_name: str, fan_in_count: int, symbols: Dict) -> AnalysisTask:
        """Create task for high usage using configured thresholds"""
        symbol_info = symbols.get(symbol_name, {})
        
        priority = self._get_priority_from_threshold(
            fan_in_count,
            self.config.get().fan_in_medium,
            self.config.get().fan_in_high,
            self.config.get().fan_in_critical
        )

        evidence = TaskEvidence(
            symbol=symbol_name,
            metric_type="fan_in",
            metric_value=fan_in_count,
            file_path=symbol_info.get("file", "unknown"),
            callers=symbol_info.get("called_by", [])[:20],
            source="rule_based" 
        )

        suggested_fix = f"""Add an abstraction layer around {symbol_name}.

This function is called by {fan_in_count} other functions. Create an interface/abstract class, then update callers to depend on the abstraction instead of the concrete function."""

        return AnalysisTask(
            id=f"task_usage_{symbol_name}_{uuid.uuid4().hex[:8]}",
            title=f"High usage of {symbol_name}",
            description=f"Function '{symbol_name}' is called by {fan_in_count} other functions. Changes here will break many dependents.",
            category=TaskCategory.HIGH_USAGE,
            priority=priority,
            status=TaskStatus.PENDING,
            evidence=[evidence],
            suggested_fix=suggested_fix
        )
    
    def generate_ai_only_tasks(self) -> List[AnalysisTask]:
        """Generate ONLY AI-discovered tasks (no rule-based)"""
        if not self.config.get().llm_enabled:
            return []
        
        return self.llm_discovery.generate_only_llm_tasks()

import json
from pathlib import Path
from typing import List, Dict, Any, Optional
from datetime import datetime

from backend.models.task import AnalysisTask, TaskStatus, TaskCategory, TaskPriority, TaskEvidence


class TaskStorage:
    """Persistent storage for tasks (JSON file-based)"""

    def __init__(self, project_id: str, data_dir: Path):
        self.project_id = project_id
        self.tasks_file = data_dir / "tasks" / f"{project_id}.json"
        self._ensure_dir()

    def _ensure_dir(self):
        """Ensure tasks directory exists"""
        self.tasks_file.parent.mkdir(parents=True, exist_ok=True)

    def save_tasks(self, tasks: List[AnalysisTask]) -> None:
        """Save all tasks to disk"""
        data = {
            "project_id": self.project_id,
            "last_updated": datetime.now().isoformat(),
            "tasks": [t.to_dict() for t in tasks]
        }
        self.tasks_file.write_text(json.dumps(data, indent=2), encoding="utf-8")

    def load_tasks(self) -> List[Dict[str, Any]]:
        """Load tasks from disk as dicts"""
        if not self.tasks_file.exists():
            return []
        data = json.loads(self.tasks_file.read_text(encoding="utf-8"))
        return data.get("tasks", [])

    def _dict_to_task(self, task_dict: Dict[str, Any]) -> AnalysisTask:
        """Convert a dictionary back to an AnalysisTask object"""
        # Reconstruct evidence objects
        evidence_objects = []
        for e in task_dict.get("evidence", []):
            evidence_objects.append(TaskEvidence(
                symbol=e["symbol"],
                metric_type=e["metric_type"],
                metric_value=e["metric_value"],
                file_path=e["file_path"],
                line_range=e.get("line_range"),
                callers=e.get("callers", []),
                callees=e.get("callees", []),
                source=e.get("source", "rule_based")
            ))
        
        # Handle created_at and updated_at
        created_at = task_dict.get("created_at")
        if isinstance(created_at, str):
            created_at = datetime.fromisoformat(created_at)
        
        updated_at = task_dict.get("updated_at")
        if isinstance(updated_at, str):
            updated_at = datetime.fromisoformat(updated_at)
        
        # Handle status - could be string or enum
        status_value = task_dict.get("status")
        if isinstance(status_value, str):
            try:
                status = TaskStatus(status_value)
            except ValueError:
                status = TaskStatus.PENDING
        else:
            status = status_value if hasattr(status_value, 'value') else TaskStatus.PENDING
        
        return AnalysisTask(
            id=task_dict["id"],
            title=task_dict["title"],
            description=task_dict["description"],
            category=TaskCategory(task_dict["category"]),
            priority=TaskPriority(task_dict["priority"]),
            status=status,
            evidence=evidence_objects,
            suggested_fix=task_dict.get("suggested_fix"),
            impact_analysis=task_dict.get("impact_analysis"),
            created_at=created_at,
            updated_at=updated_at,
            notes=task_dict.get("notes")
        )

    def update_task_status(self, task_id: str, status: TaskStatus, notes: Optional[str] = None) -> bool:
        """Update a single task's status"""
        tasks_dicts = self.load_tasks()
        
        for task_dict in tasks_dicts:
            if task_dict["id"] == task_id:
                task_dict["status"] = status.value if hasattr(status, 'value') else str(status)
                task_dict["updated_at"] = datetime.now().isoformat()
                if notes:
                    task_dict["notes"] = notes
                
                # Convert all dicts back to AnalysisTask objects
                analysis_tasks = []
                for td in tasks_dicts:
                    analysis_tasks.append(self._dict_to_task(td))
                
                self.save_tasks(analysis_tasks)
                return True
        return False
