// AI Refactoring Agent - No conflicts with task_board.js

const AI_REFACTOR_STORAGE_KEY = 'ai_refactor_suggestions';

class AIRefactorAgent {
    constructor() {
        this.apiBase = '/api/refactor';
        this.suggestions = [];
        this.isAnalyzing = false;
        this.init();
    }

    init() {
        this.loadSavedSuggestions();
        this.attachEventListeners();
        this.updateBadgeCount();
    }

    attachEventListeners() {
        const analyzeBtn = document.getElementById('analyzeCodebaseBtn');
        if (analyzeBtn) {
            analyzeBtn.addEventListener('click', () => this.analyzeCodebase());
        }

        const refreshBtn = document.getElementById('refreshAISuggestionsBtn');
        if (refreshBtn) {
            refreshBtn.addEventListener('click', () => this.loadSavedSuggestions());
        }

        const clearBtn = document.getElementById('clearAISuggestionsBtn');
        if (clearBtn) {
            clearBtn.addEventListener('click', () => this.clearSuggestions());
        }
    }

    async analyzeCodebase() {
        if (this.isAnalyzing) return;

        this.isAnalyzing = true;
        this.showStatus('🤖 AI is analyzing your codebase for refactorings...');

        try {
            const functions = await this.getAllFunctions();

            this.showStatus(`📊 Found ${functions.length} functions. Analyzing...`);

            const suggestions = [];
            const functionsToAnalyze = functions.slice(0, 15);

            for (let i = 0; i < functionsToAnalyze.length; i++) {
                const func = functionsToAnalyze[i];
                this.showStatus(`🔍 Analyzing ${i + 1}/${functionsToAnalyze.length}: ${func.name}...`);

                try {
                    const response = await fetch(`${this.apiBase}/suggest/${encodeURIComponent(func.name)}`, {
                        method: 'POST'
                    });

                    if (response.ok) {
                        const data = await response.json();
                        if (data.suggestions && data.suggestions.length > 0) {
                            suggestions.push({
                                function_name: func.name,
                                file: func.file,
                                type: func.type,
                                parent: func.parent,
                                suggestions: data.suggestions,
                                analyzed_at: new Date().toISOString()
                            });
                        }
                    }
                } catch (e) {
                    console.error(`Error analyzing ${func.name}:`, e);
                }

                await new Promise(r => setTimeout(r, 500));
            }

            this.suggestions = suggestions;
            this.saveSuggestions();
            this.renderSuggestions();
            this.updateBadgeCount();
            this.showStatus(`✅ Analysis complete! Found ${suggestions.length} functions with suggestions.`, 'success');

        } catch (error) {
            console.error('Analysis failed:', error);
            this.showStatus('❌ Analysis failed. Please try again.', 'error');
        } finally {
            this.isAnalyzing = false;
            setTimeout(() => this.hideStatus(), 3000);
        }
    }

    async getAllFunctions() {
        try {
            const response = await fetch(`${this.apiBase}/search/functions?query=&limit=100`);
            const data = await response.json();
            return data.functions || [];
        } catch (error) {
            console.error('Error fetching functions:', error);
            return [];
        }
    }

    showStatus(message, type = 'info') {
        const statusDiv = document.getElementById('aiStatus');
        const messageDiv = statusDiv?.querySelector('.status-message');

        if (statusDiv) {
            statusDiv.style.display = 'block';
            if (messageDiv) {
                messageDiv.textContent = message;
                messageDiv.className = `status-message status-${type}`;
            }
        }
    }

    hideStatus() {
        const statusDiv = document.getElementById('aiStatus');
        if (statusDiv) {
            setTimeout(() => {
                statusDiv.style.display = 'none';
            }, 2000);
        }
    }

    saveSuggestions() {
        localStorage.setItem(AI_REFACTOR_STORAGE_KEY, JSON.stringify(this.suggestions));
        localStorage.setItem(`${AI_REFACTOR_STORAGE_KEY}_timestamp`, new Date().toISOString());
    }

    loadSavedSuggestions() {
        const saved = localStorage.getItem(AI_REFACTOR_STORAGE_KEY);
        if (saved) {
            this.suggestions = JSON.parse(saved);
            this.renderSuggestions();
            this.updateBadgeCount();
            const timestamp = localStorage.getItem(`${AI_REFACTOR_STORAGE_KEY}_timestamp`);
            if (timestamp) {
                this.showStatus(`📋 Loaded ${this.suggestions.length} suggestions from ${new Date(timestamp).toLocaleString()}`, 'info');
                setTimeout(() => this.hideStatus(), 2000);
            }
        } else {
            this.renderEmpty();
        }
    }

    clearSuggestions() {
        if (confirm('Clear all AI refactoring suggestions?')) {
            this.suggestions = [];
            localStorage.removeItem(AI_REFACTOR_STORAGE_KEY);
            localStorage.removeItem(`${AI_REFACTOR_STORAGE_KEY}_timestamp`);
            this.renderEmpty();
            this.updateBadgeCount();
            this.showStatus('🗑️ All suggestions cleared', 'info');
            setTimeout(() => this.hideStatus(), 1500);
        }
    }

    updateBadgeCount() {
        const badge = document.getElementById('aiGeneratedCount');
        if (badge) {
            const totalSuggestions = this.suggestions.reduce((sum, item) => sum + item.suggestions.length, 0);
            badge.textContent = totalSuggestions;
        }
    }

    renderEmpty() {
        const container = document.getElementById('aiSuggestionsList');
        if (container) {
            container.innerHTML = `
                <div class="info-message">
                    💡 No refactoring suggestions yet.<br>
                    Click "Analyze Codebase" to get AI-powered recommendations.
                </div>
            `;
        }
    }

    renderSuggestions() {
        const container = document.getElementById('aiSuggestionsList');
        if (!container) return;

        if (!this.suggestions || this.suggestions.length === 0) {
            this.renderEmpty();
            return;
        }

        let totalSuggestions = 0;
        let html = '';

        for (const item of this.suggestions) {
            for (const suggestion of item.suggestions) {
                totalSuggestions++;
                html += this.renderSuggestionCard(item, suggestion);
            }
        }

        if (totalSuggestions === 0) {
            this.renderEmpty();
            return;
        }

        container.innerHTML = `
            <div class="suggestions-header">
                <span>🎯 Found ${totalSuggestions} refactoring suggestions</span>
            </div>
            ${html}
        `;

        this.attachCardHandlers();
        this.updateBadgeCount();
    }

    renderSuggestionCard(item, suggestion) {
        const difficultyClass = (suggestion.difficulty || 'medium').toLowerCase();
        const typeIcon = this.getTypeIcon(suggestion.type);
        const typeName = (suggestion.type || 'refactoring').replace(/_/g, ' ').toUpperCase();

        return `
            <div class="ai-suggestion-card" data-function="${item.function_name}" data-suggestion-type="${suggestion.type}">
                <div class="suggestion-header">
                    <div class="suggestion-type ${difficultyClass}">
                        ${typeIcon} ${typeName}
                    </div>
                    <div class="suggestion-difficulty ${difficultyClass}">
                        📊 ${suggestion.difficulty || 'Medium'} Difficulty
                    </div>
                </div>
                <div class="suggestion-function">
                    <strong>🎯 Function:</strong> ${item.function_name}
                    ${item.parent ? `<span class="class-name">(${item.parent})</span>` : ''}
                </div>
                <div class="suggestion-file">
                    <strong>📁 File:</strong> ${item.file?.split('/').pop() || 'Unknown'}
                </div>
                <div class="suggestion-reason">
                    <strong>💡 Reason:</strong> ${suggestion.reason || 'Improve code quality and maintainability'}
                </div>
                ${suggestion.affected_lines ? `
                <div class="suggestion-lines">
                    <strong>📍 Lines:</strong> ${suggestion.affected_lines}
                </div>
                ` : ''}
                <div class="suggestion-actions">
                    <button class="btn-preview" data-function="${item.function_name}" data-type="${suggestion.type}" data-suggestion='${this.escapeJson(JSON.stringify(suggestion))}'>
                        👁️ Preview Changes
                    </button>
                    <button class="btn-generate-plan" data-function="${item.function_name}" data-type="${suggestion.type}" data-suggestion='${this.escapeJson(JSON.stringify(suggestion))}'>
                        🔨 Generate Plan
                    </button>
                </div>
            </div>
        `;
    }

    escapeJson(str) {
        return str.replace(/'/g, "&#39;").replace(/"/g, '&quot;');
    }

    getTypeIcon(type) {
        const icons = {
            'extract_method': '📦',
            'inline_function': '🔄',
            'add_parameter': '➕',
            'remove_parameter': '➖',
            'rename_function': '✏️',
            'simplify_logic': '🧹'
        };
        return icons[type] || '🔧';
    }

    attachCardHandlers() {
        document.querySelectorAll('#aiSuggestionsList .btn-preview').forEach(btn => {
            btn.removeEventListener('click', this.handlePreview);
            btn.addEventListener('click', this.handlePreview.bind(this));
        });

        document.querySelectorAll('#aiSuggestionsList .btn-generate-plan').forEach(btn => {
            btn.removeEventListener('click', this.handleGenerate);
            btn.addEventListener('click', this.handleGenerate.bind(this));
        });
    }

    async handlePreview(event) {
        const btn = event.currentTarget;
        const functionName = btn.dataset.function;
        const type = btn.dataset.type;
        const suggestion = JSON.parse(btn.dataset.suggestion);
        await this.previewRefactoring(functionName, type, suggestion);
    }

    async handleGenerate(event) {
        const btn = event.currentTarget;
        const functionName = btn.dataset.function;
        const type = btn.dataset.type;
        const suggestion = JSON.parse(btn.dataset.suggestion);
        await this.generateAndShowPlan(functionName, type, suggestion);
    }

    async previewRefactoring(functionName, type, suggestion) {
        const modal = document.getElementById('taskModal');
        const modalBody = document.getElementById('modalBody');
        if (!modal || !modalBody) return;

        modalBody.innerHTML = '<div class="loading">Generating preview...</div>';
        modal.style.display = 'block';

        try {
            let plan = null;

            switch (type) {
                case 'extract_method':
                    const lines = suggestion.affected_lines?.split('-') || ['0', '0'];
                    plan = await this.callExtractMethod(functionName, parseInt(lines[0]), parseInt(lines[1]), suggestion.suggested_name || '_extracted');
                    break;
                case 'inline_function':
                    plan = await this.callInlineFunction(functionName);
                    break;
                default:
                    modalBody.innerHTML = '<div class="info-message">Preview not available for this refactoring type yet</div>';
                    return;
            }

            if (plan && plan.changes) {
                this.showPlanPreview(plan);
            } else {
                modalBody.innerHTML = '<div class="info-message">Failed to generate preview</div>';
            }
        } catch (error) {
            modalBody.innerHTML = `<div class="info-message">Error: ${error.message}</div>`;
        }
    }

    async generateAndShowPlan(functionName, type, suggestion) {
        const modal = document.getElementById('taskModal');
        const modalBody = document.getElementById('modalBody');
        if (!modal || !modalBody) return;

        modalBody.innerHTML = '<div class="loading">Generating refactoring plan...</div>';
        modal.style.display = 'block';

        try {
            let plan = null;

            switch (type) {
                case 'extract_method':
                    const lines = suggestion.affected_lines?.split('-') || ['0', '0'];
                    plan = await this.callExtractMethod(functionName, parseInt(lines[0]), parseInt(lines[1]), suggestion.suggested_name || '_extracted');
                    break;
                case 'inline_function':
                    plan = await this.callInlineFunction(functionName);
                    break;
                default:
                    modalBody.innerHTML = '<div class="info-message">Plan generation not available for this refactoring type yet</div>';
                    return;
            }

            if (plan) {
                this.showPlanPreview(plan);
            } else {
                modalBody.innerHTML = '<div class="info-message">Failed to generate plan</div>';
            }
        } catch (error) {
            modalBody.innerHTML = `<div class="info-message">Error: ${error.message}</div>`;
        }
    }

    async callExtractMethod(functionName, startLine, endLine, newMethodName) {
        const response = await fetch(`${this.apiBase}/plan/extract-method`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                function_name: functionName,
                start_line: startLine,
                end_line: endLine,
                new_method_name: newMethodName
            })
        });
        return response.json();
    }

    async callInlineFunction(functionName) {
        const response = await fetch(`${this.apiBase}/plan/inline-function`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ function_name: functionName })
        });
        return response.json();
    }

    showPlanPreview(plan) {
        const modalBody = document.getElementById('modalBody');
        if (!modalBody) return;

        let changesHtml = '';
        for (const change of plan.changes) {
            changesHtml += `
                <div class="plan-change">
                    <div class="change-file">📄 ${(change.file_path || '').split('/').pop()}</div>
                    <div class="change-description">${change.description || 'Code change'}</div>
                    <div class="diff-view">
                        <div class="diff-before">- ${(change.before || '').split('\n')[0].substring(0, 100)}</div>
                        <div class="diff-after">+ ${(change.after || '').split('\n')[0].substring(0, 100)}</div>
                    </div>
                    <button class="copy-change-btn" data-before="${this.escapeHtml(change.before || '')}" data-after="${this.escapeHtml(change.after || '')}">
                        📋 Copy Code Change
                    </button>
                </div>
            `;
        }

        modalBody.innerHTML = `
            <div class="plan-details">
                <h3>Refactoring Plan</h3>
                <div class="plan-meta">
                    <div>🎯 Target: ${plan.target_function || 'Unknown'}</div>
                    <div>📝 Type: ${plan.refactor_type || 'Unknown'}</div>
                    <div>🎯 Confidence: ${Math.round((plan.confidence || 0) * 100)}%</div>
                    <div>${plan.requires_review ? '⚠️ Requires Review' : '✅ Auto-approved'}</div>
                </div>
                <div class="plan-changes">
                    <h4>Changes (${plan.changes?.length || 0} files):</h4>
                    ${changesHtml}
                </div>
            </div>
        `;

        document.querySelectorAll('.copy-change-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                const text = `# BEFORE:\n${this.unescapeHtml(btn.dataset.before)}\n\n# AFTER:\n${this.unescapeHtml(btn.dataset.after)}`;
                navigator.clipboard.writeText(text);
                btn.textContent = '✅ Copied!';
                setTimeout(() => {
                    btn.textContent = '📋 Copy Code Change';
                }, 2000);
            });
        });
    }

    escapeHtml(str) {
        if (!str) return '';
        return str.replace(/[&<>]/g, function (m) {
            if (m === '&') return '&amp;';
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }

    unescapeHtml(str) {
        if (!str) return '';
        return str.replace(/&amp;/g, '&').replace(/&lt;/g, '<').replace(/&gt;/g, '>');
    }
}

document.addEventListener("DOMContentLoaded", () => {

    const STORAGE_KEY = "idm_dev_agent_chat";

    const chatWindow =
        document.getElementById("chat-window");

    const input =
        document.getElementById("chat-input");

    const button =
        document.getElementById("send-button");

    const newChatButton =
        document.getElementById("new-chat-button");

    let history = [];

    function saveHistory() {

        localStorage.setItem(

            STORAGE_KEY,

            JSON.stringify(history)

        );

    }

    function scrollBottom() {

        chatWindow.scrollTop =
            chatWindow.scrollHeight;

    }

    function renderMessage(role, text, metadata = null) {

        const wrapper =
            document.createElement("div");

        wrapper.className =
            role === "user"
                ? "user-message"
                : "assistant-message";

        const avatar =
            document.createElement("div");

        avatar.className = "avatar";

        avatar.textContent =
            role === "user"
                ? "👤"
                : "🤖";

        const bubble =
            document.createElement("div");

        bubble.className = "bubble";

        if (role === "assistant") {

            bubble.innerHTML =
                marked.parse(text);

            bubble.querySelectorAll("pre code")
                .forEach(block => {

                    hljs.highlightElement(block);

                });

            const copy =
                document.createElement("button");

            copy.className =
                "copy-button";

            copy.textContent = "Copy";

            copy.onclick = () => {

                navigator.clipboard.writeText(text).catch(console.error);

                copy.textContent = "Copied!";

                setTimeout(() => {

                    copy.textContent = "Copy";

                }, 1500);

            };

            bubble.prepend(copy);

            if (metadata && metadata.metrics) {

                const details =
                    document.createElement("details");

                details.className = "diagnostics";

                const summary =
                    document.createElement("summary");

                summary.textContent =
                    "Diagnostics";

                details.appendChild(summary);

                const table =
                    document.createElement("table");

                table.innerHTML = `

                        <tr>

                        <td class="metric-label">Intent</td>

                        <td>${metadata.intent}</td>

                        </tr>

                        <tr>

                        <td class="metric-label">Resolved Symbol</td>

                        <td>${metadata.symbol}</td>

                        </tr>

                        <tr>

                        <td class="metric-label">Context Size</td>

                        <td>${metadata.metrics.context_chars} chars</td>

                        </tr>

                        <tr>

                        <td class="metric-label">Prompt Size</td>

                        <td>${metadata.metrics.prompt_chars} chars</td>

                        </tr>

                        <tr>

                        <td class="metric-label">Model</td>

                        <td>${metadata.metrics.model}</td>

                        </tr>

                        <tr>

                        <td class="metric-label">LLM Time</td>

                        <td>${metadata.metrics.llm_ms} ms</td>

                        </tr>

                        <tr>

                        <td class="metric-label">Total Time</td>

                        <td>${metadata.metrics.total_ms} ms</td>

                        </tr>

                        `;

                details.appendChild(table);

                bubble.appendChild(details);

            }

        }
        else {

            bubble.textContent = text;

        }

        wrapper.appendChild(avatar);

        wrapper.appendChild(bubble);

        chatWindow.appendChild(wrapper);

        scrollBottom();

    }

    function redrawConversation() {

        chatWindow.innerHTML = "";

        if (history.length === 0) {

            renderMessage(

                "assistant",

                "# Welcome\n\nAsk me anything about your project."

            );

            return;

        }

        history.forEach(msg => {

            renderMessage(

                msg.role,

                msg.text,

                msg.metadata || null

            );

        });

    }

    async function sendMessage() {

        const message =
            input.value.trim();

        if (!message)
            return;

        history.push({

            role: "user",

            text: message

        });

        saveHistory();

        renderMessage(

            "user",

            message

        );

        input.value = "";

        const loadingWrapper =
            document.createElement("div");

        loadingWrapper.className =
            "assistant-message";

        loadingWrapper.innerHTML =

            '<div class="avatar">🤖</div><div class="bubble">Thinking...</div>';

        chatWindow.appendChild(

            loadingWrapper

        );

        scrollBottom();

        try {

            const response =
                await fetch(

                    "/agent/chat",

                    {

                        method: "POST",

                        headers: {

                            "Content-Type":

                                "application/json"

                        },

                        body: JSON.stringify({

                            message

                        })

                    }

                );

            const data =
                await response.json();

            loadingWrapper.remove();

            history.push({

                role: "assistant",

                text: data.answer,

                metadata: {

                    intent: data.intent,

                    symbol: data.symbol,

                    metrics: data.metrics

                }

            });

            saveHistory();

            renderMessage(
                "assistant",
                data.answer,
                data
            );

        }
        catch {

            loadingWrapper.remove();

            history.push({

                role: "assistant",

                text: "**Error contacting agent.**"

            });

            saveHistory();

            renderMessage(

                "assistant",

                "**Error contacting agent.**"

            );

        }

    }

    button.onclick = sendMessage;

    input.addEventListener(

        "keydown",

        e => {

            if (

                e.key === "Enter"

                &&

                !e.shiftKey

            ) {

                e.preventDefault();

                sendMessage();

            }

        }

    );

    newChatButton.onclick = () => {

        if (

            confirm(

                "Start a new conversation?"

            )

        ) {

            history = [];

            saveHistory();

            redrawConversation();

        }

    };

    const stored =
        localStorage.getItem(
            STORAGE_KEY
        );

    if (stored) {

        try {
            history = JSON.parse(stored);
        }
        catch {
            history = [];
        }

    }

    redrawConversation();

});

    attachEventListeners() {

                clearTimeout(debounceTimer);

                            <div class="diff-after">+ ${change.after.split('\n')[0].substring(0, 100)}...</div>
                        </div>
                        <button class="copy-change" data-before="${this.escapeHtml(change.before)}" data-after="${this.escapeHtml(change.after)}">
                            📋 Copy to Clipboard
                        </button>
                    </div>
                `).join('')}
            </div>
        `;
        previewDiv.style.display = 'block';

        document.querySelectorAll('.copy-change').forEach(btn => {

        });

    async copyChange(data) {
        const text = `# BEFORE:\n${this.unescapeHtml(data.before)}\n\n# AFTER:\n${this.unescapeHtml(data.after)}`;
        await navigator.clipboard.writeText(text);
        
        event.target.textContent = '✅ Copied!';
        setTimeout(() => {
            event.target.textContent = '📋 Copy to Clipboard';
        }, 2000);
    }

    escapeHtml(str) {
        return str.replace(/[&<>]/g, function(m) {
            if (m === '&') return '&amp;';
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }

    unescapeHtml(str) {
        return str.replace(/&amp;/g, '&').replace(/&lt;/g, '<').replace(/&gt;/g, '>');
    }
}

function filterTasks() {
    if (!allTasks) return;
    
    // Safely get filter values with defaults
    const status = statusFilter ? statusFilter.value : 'all';
    const priority = priorityFilter ? priorityFilter.value : 'all';
    const category = categoryFilter ? categoryFilter.value : 'all';
    const source = sourceFilter ? sourceFilter.value : 'all';
    const search = searchFilter ? searchFilter.value.toLowerCase() : '';

    let filtered = [...allTasks];

    if (status !== 'all') {
        filtered = filtered.filter(t => t.status === status);
    }
    if (priority !== 'all') {
        filtered = filtered.filter(t => t.priority === priority);
    }
    if (category !== 'all') {
        filtered = filtered.filter(t => t.category === category);
    }
    if (source !== 'all') {
        filtered = filtered.filter(t => t.evidence?.[0]?.source === source);
    }
    if (search) {
        filtered = filtered.filter(t => t.title.toLowerCase().includes(search));
    }

    renderTasks(filtered);
}

async function generateTasks() {
    if (!generateBtn) return;
    
    generateBtn.disabled = true;
    generateBtn.textContent = '⏳ Starting generation...';

    let includeLlm = true;
    try {
        const configResponse = await fetch(`${API_BASE}/config`);
        const config = await configResponse.json();
        includeLlm = config.llm_enabled;
    } catch (e) {
        console.error('Error checking LLM config:', e);
    }

    try {
        const startResponse = await fetch(`${API_BASE}/tasks/generate-async?include_llm=${includeLlm}`, {
            method: 'POST'
        });
        const startResult = await startResponse.json();

        const jobId = startResult.job_id;
        generateBtn.textContent = '⏳ Generating tasks (this may take a few minutes)...';

        let completed = false;
        let pollCount = 0;
        const maxPolls = 120;

        while (!completed && pollCount < maxPolls) {
            await new Promise(resolve => setTimeout(resolve, 1000));

            const statusResponse = await fetch(`${API_BASE}/tasks/generate-status/${jobId}`);
            const status = await statusResponse.json();

            if (status.status === 'completed') {
                completed = true;
                alert(`Task generation complete!\nRule-based: ${status.result.rule_based_tasks}\nAI-discovered: ${status.result.llm_discovered_tasks}`);
                loadTasks();
                loadStats();
            } else if (status.status === 'failed') {
                completed = true;
                alert(`Task generation failed: ${status.error}`);
            }

            pollCount++;
        }

        if (!completed) {
            alert('Task generation is taking longer than expected. Check back later.');
        }

    } catch (error) {
        console.error('Error generating tasks:', error);
        alert('Error starting task generation');
    } finally {
        generateBtn.disabled = false;
        generateBtn.textContent = '🔄 Generate Rule-Based Tasks';
    }
}

function escapeHtml(text) {
    if (!text) return '';
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}

async function generateAITasks() {
    if (!generateAIBtn) return;
    
    generateAIBtn.disabled = true;
    generateAIBtn.textContent = '🤖 Starting AI generation...';
    
    try {
        const startResponse = await fetch(`${API_BASE}/tasks/generate-ai-only-async`, {
            method: 'POST'
        });
        const startResult = await startResponse.json();
        
        const jobId = startResult.job_id;
        generateAIBtn.textContent = '🤖 AI generating tasks (may take a few minutes)...';
        
        let completed = false;
        let pollCount = 0;
        const maxPolls = 120;
        
        while (!completed && pollCount < maxPolls) {
            await new Promise(resolve => setTimeout(resolve, 1000));
            
            const statusResponse = await fetch(`${API_BASE}/tasks/generate-status/${jobId}`);
            const status = await statusResponse.json();
            
            if (status.status === 'completed') {
                completed = true;
                alert(`AI task generation complete!\nAdded ${status.result.ai_tasks_added} new AI-discovered tasks\nTotal tasks: ${status.result.total_tasks}`);
                loadTasks();
                loadStats();
                if (sourceFilter) sourceFilter.value = 'llm';
                filterTasks();
            } else if (status.status === 'failed') {
                completed = true;
                alert(`AI task generation failed: ${status.error}`);
            }
            
            pollCount++;
        }
        
        if (!completed) {
            alert('AI task generation is taking longer than expected. Check back later.');
        }
        
    } catch (error) {
        console.error('Error generating AI tasks:', error);
        alert('Error starting AI task generation');
    } finally {
        generateAIBtn.disabled = false;
        generateAIBtn.textContent = '🤖 Generate AI Tasks Only';
    }
}

        let node, link;
        async function loadGraph() {
            const res = await fetch('/graph/data');
            const data = await res.json();

            const nodesMap = {};
            const links = [];

            // build nodes + links
            for (const [source, targets] of Object.entries(data.calls)) {
                nodesMap[source] = true;

                for (const t of targets) {
                    nodesMap[t] = true;
                    links.push({ source, target: t });
                }
            }

            const nodes = Object.keys(nodesMap).map(id => ({ id }));

            const svg = d3.select("svg");
            const width = window.innerWidth;
            const height = window.innerHeight;

            const simulation = d3.forceSimulation(nodes)
                .force("link", d3.forceLink(links).id(d => d.id).distance(80))
                .force("charge", d3.forceManyBody().strength(-200))
                .force("center", d3.forceCenter(width / 2, height / 2));


            link = svg.append("g")
                .selectAll("line")
                .data(links)
                .enter()
                .append("line")
                .attr("class", "link");

            node = svg.append("g")
                .selectAll("g")
                .data(nodes)
                .enter()
                .append("g")
                .attr("class", "node")
                .call(d3.drag()
                    .on("start", dragstarted)
                    .on("drag", dragged)
                    .on("end", dragended)
                );

            node.append("circle")
                .attr("r", 8)
                .on("click", (event, d) => {
                    alert(d.id);
                });

            node.append("text")
                .attr("x", 10)
                .attr("y", 3)
                .text(d => d.id);

            simulation.on("tick", () => {
                link
                    .attr("x1", d => d.source.x)
                    .attr("y1", d => d.source.y)
                    .attr("x2", d => d.target.x)
                    .attr("y2", d => d.target.y);

                node.attr("transform", d => `translate(${d.x},${d.y})`);
            });

            function dragstarted(event, d) {
                if (!event.active) simulation.alphaTarget(0.3).restart();
                d.fx = d.x;
                d.fy = d.y;
            }

            function dragged(event, d) {
                d.fx = event.x;
                d.fy = event.y;
            }

            function dragended(event, d) {
                if (!event.active) simulation.alphaTarget(0);
                d.fx = null;
                d.fy = null;
            }
        }

        loadGraph();
        document.getElementById("searchBox").addEventListener("input", function (e) {
            const value = e.target.value.trim().toLowerCase();

            node.style("opacity", d => {
                if (!value) return 1;
                return d.id.toLowerCase().includes(value) ? 1 : 0.1;
            });

            link.style("opacity", d => {
                if (!value) return 1;
                return (
                    d.source.id.toLowerCase().includes(value) ||
                    d.target.id.toLowerCase().includes(value)
                ) ? 1 : 0.05;
            });
        });

        // Reset and lock elements during active execution
        inputEl.value = '';

            const params = new URLSearchParams({ question: question, symbol_name: symbol });
            const response = await fetch(`http://127.0.0.1:8000/api/agent/stream?${params.toString()}`);

            const reader = response.body.getReader();
            const decoder = new TextDecoder('utf-8');

            while (true) {
                const { value, done } = await reader.read();
                if (done) break;

                const token = decoder.decode(value, { stream: true });
                contentTarget.textContent += token;
                chatContainer.scrollTop = chatContainer.scrollHeight;
            }
        } catch (error) {
            contentTarget.textContent = `[Streaming Pipeline Execution Failure: ${error.message}]`;
        }

        sendBtn.disabled = false;
        inputEl.focus();
    });
